020package org.elasticsearch.websocket;
021
022import java.io.IOException;
023import java.net.URI;
024import java.util.concurrent.TimeUnit;
025import org.elasticsearch.common.logging.ESLogger;
026import org.elasticsearch.common.logging.ESLoggerFactory;
027import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
028import org.elasticsearch.websocket.client.WebSocketActionListener;
029import org.elasticsearch.websocket.client.WebSocketClient;
030import org.elasticsearch.websocket.client.WebSocketClientFactory;
031import org.elasticsearch.websocket.client.WebSocketClientRequest;
032import org.elasticsearch.websocket.http.netty.client.NettyWebSocketClientFactory;
033import org.jboss.netty.handler.codec.http.websocketx.CloseWebSocketFrame;
034import org.jboss.netty.handler.codec.http.websocketx.WebSocketFrame;
035import org.testng.annotations.Test;
036
037public class PublishSubscribeRequestTest {
038    
039    private final static ESLogger logger = ESLoggerFactory.getLogger("test");    
040    
041    public void testOneClient() {
042        try {
043            final String subscriberId = "oneclient";
044            final String topic = "oneclienttest";
045            final URI uri = new URI("ws://localhost:9400/websocket");
046            final WebSocketClientFactory clientFactory = new NettyWebSocketClientFactory();
047            
048            final WebSocketClientRequest subscribe = clientFactory.newRequest()
049                    .type("subscribe").data(jsonBuilder().startObject()
050                    .field("subscriber", "singleclient")
051                    .field("topic", topic).endObject());
052            
053            final WebSocketClientRequest publish = clientFactory.newRequest()
054                    .type("publish").data(jsonBuilder().startObject()
055                    .field("message", "Hello World")
056                    .field("topic", topic).endObject());
057            
058            WebSocketClient client = clientFactory.newClient(uri,
059                    new WebSocketActionListener.Adapter() {
060                        @Override
061                        public void onConnect(WebSocketClient client) throws IOException {
062                                logger.info("sending subscribe command, channel = {}", client.channel());
063                                subscribe.send(client);
064                                logger.info("sending publish command (to ourselves), channel = {}", client.channel());
065                                publish.send(client);
066                        }
067
068                        @Override
069                        public void onMessage(WebSocketClient client, WebSocketFrame frame) {
070                            logger.info("frame received: " + frame);
071                        }
072                    });
073            client.connect().await(1000, TimeUnit.MILLISECONDS);            
074            Thread.sleep(1000);
075            client.send(new CloseWebSocketFrame());
076            Thread.sleep(1000);
077            client.disconnect();
078            
079            clientFactory.shutdown();
080            
081        } catch (Exception e) {
082            logger.error(e.getMessage(), e);
083        }        
084    }
085
086    @Test
087    public void testTwoClients() {
088        try {
089            final String subscriberId = "twoclients";
090            final String topic = "twoclienttest";
091            final URI uri = new URI("ws://localhost:9400/websocket");
092            final WebSocketClientFactory clientFactory = new NettyWebSocketClientFactory();
093
094            final WebSocketClientRequest subscribe = clientFactory.newRequest()
095                    .type("subscribe").data(jsonBuilder().startObject()
096                    .field("subscriber", "doubleclient")
097                    .field("topic", topic).endObject());
098            
099            final WebSocketClientRequest publish = clientFactory.newRequest()
100                    .type("publish").data(jsonBuilder().startObject()
101                    .field("message", "Hi there, I'm another client")
102                    .field("topic", topic).endObject());
103            
104            // open first client
105            WebSocketClient subscribingClient = clientFactory.newClient(uri,
106                    new WebSocketActionListener.Adapter() {
107                        @Override
108                        public void onConnect(WebSocketClient client) {
109                            try {
110                                logger.info("sending subscribe command, channel {}", client.channel());
111                                subscribe.send(client);
112                            } catch (Exception e) {
113                            }
114                        }
115
116                        @Override
117                        public void onMessage(WebSocketClient client, WebSocketFrame frame) {
118                            logger.info("frame {} received for subscribing client {}", frame, client.channel());
119                        }
120
121                    });
122            
123            // open two client
124            WebSocketClient publishingClient = clientFactory.newClient(uri,
125                    new WebSocketActionListener.Adapter() {
126                        @Override
127                        public void onConnect(WebSocketClient client) {
128                            try {
129                                logger.info("sending publish command, channel = {}", client.channel());
130                                publish.send(client);
131                            } catch (Exception e) {
132                            }
133                        }
134
135                        @Override
136                        public void onMessage(WebSocketClient client, WebSocketFrame frame) {
137                            logger.info("frame {} received for publishing client {}", frame, client.channel());
138                        }
139
140                    });
141            
142            // connect both clients to node
143            subscribingClient.connect().await(1000, TimeUnit.MILLISECONDS);            
144            publishingClient.connect().await(1000, TimeUnit.MILLISECONDS);            
145            
146            // wait for publish/subscribe
147            Thread.sleep(1000);
148            
149            // close first client
150            publishingClient.send(new CloseWebSocketFrame());
151            publishingClient.disconnect();
152
153            // close second client
154            subscribingClient.send(new CloseWebSocketFrame());
155            subscribingClient.disconnect();
156            
157            Thread.sleep(1000);
158            
159            clientFactory.shutdown();
160            
161        } catch (Exception e) {
162            logger.error(e.getMessage(), e);
163        }        
164    }
