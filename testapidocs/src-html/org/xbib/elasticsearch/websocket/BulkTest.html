002    package org.xbib.elasticsearch.websocket;
003    
004    import java.util.concurrent.TimeUnit;
005    
006    import org.jboss.netty.handler.codec.http.websocketx.CloseWebSocketFrame;
007    import org.jboss.netty.handler.codec.http.websocketx.WebSocketFrame;
008    
009    import org.elasticsearch.common.logging.ESLogger;
010    import org.elasticsearch.common.logging.ESLoggerFactory;
011    
012    import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
013    
014    import org.junit.Test;
015    import org.xbib.elasticsearch.websocket.client.WebSocketActionListener;
016    import org.xbib.elasticsearch.websocket.client.WebSocketClient;
017    import org.xbib.elasticsearch.websocket.client.WebSocketClientFactory;
018    import org.xbib.elasticsearch.websocket.helper.AbstractNodeTestHelper;
019    import org.xbib.elasticsearch.websocket.http.netty.client.NettyWebSocketClientFactory;
020    
021    public class BulkTest extends AbstractNodeTestHelper {
022    
023        private final static ESLogger logger = ESLoggerFactory.getLogger("test");
024    
025        @Test
026        public void testBulk() {
027            try {
028                final WebSocketClientFactory clientFactory = new NettyWebSocketClientFactory();
029                WebSocketClient client = clientFactory.newClient(
030                        getAddressOfNode("1"),
031                        new WebSocketActionListener.Adapter() {
032                            @Override
033                            public void onConnect(WebSocketClient client) {
034                                try {
035                                    logger.info("sending some index requests (longer than a single bulk size)");
036                                    for (int i = 0; i < 250; i++) {
037                                        clientFactory.indexRequest()
038                                                .data(jsonBuilder()                                            
039                                                .startObject()
040                                                .field("index", "test")
041                                                .field("type", "test")
042                                                .field("id", Integer.toString(i))
043                                                .startObject("data")
044                                                .field("field1", "value" + i)
045                                                .field("field2", "value" + i)
046                                                .endObject()
047                                                .endObject())
048                                                .send(client);
049                                    }
050                                    // more bulks could be added here ...
051                                    logger.info("at the end, let us flush the bulk");
052                                    clientFactory.flushRequest().send(client);
053                                } catch (Exception e) {
054                                    onError(e);
055                                }
056                            }
057    
058                            @Override
059                            public void onDisconnect(WebSocketClient client) {
060                                logger.info("disconnected");
061                            }
062    
063                            @Override
064                            public void onMessage(WebSocketClient client, WebSocketFrame frame) {
065                                logger.info("frame received: {}", frame);
066                            }
067    
068                            @Override
069                            public void onError(Throwable t) {
070                                logger.error(t.getMessage(), t);
071                            }
072                        });
073                client.connect().await(1000, TimeUnit.MILLISECONDS);
074                Thread.sleep(1000);
075                logger.info("closing bulk client");
076                client.send(new CloseWebSocketFrame());
077                Thread.sleep(1000);
078                client.disconnect();
079                clientFactory.shutdown();
080            } catch (Exception e) {
081                logger.error(e.getMessage(), e);
082            }
083    
084        }
085    }
