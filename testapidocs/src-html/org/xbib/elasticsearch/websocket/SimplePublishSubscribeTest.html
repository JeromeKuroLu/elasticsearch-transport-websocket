002    package org.xbib.elasticsearch.websocket;
003    
004    import java.util.concurrent.TimeUnit;
005    
006    import org.jboss.netty.handler.codec.http.websocketx.CloseWebSocketFrame;
007    import org.jboss.netty.handler.codec.http.websocketx.TextWebSocketFrame;
008    import org.jboss.netty.handler.codec.http.websocketx.WebSocketFrame;
009    
010    import org.elasticsearch.common.logging.ESLogger;
011    import org.elasticsearch.common.logging.ESLoggerFactory;
012    
013    import org.junit.Test;
014    import org.xbib.elasticsearch.websocket.client.WebSocketActionListener;
015    import org.xbib.elasticsearch.websocket.client.WebSocketClient;
016    import org.xbib.elasticsearch.websocket.client.WebSocketClientFactory;
017    import org.xbib.elasticsearch.websocket.helper.AbstractNodeTestHelper;
018    import org.xbib.elasticsearch.websocket.http.netty.client.NettyWebSocketClientFactory;
019    
020    public class SimplePublishSubscribeTest extends AbstractNodeTestHelper {
021    
022        private final static ESLogger logger = ESLoggerFactory.getLogger("test");
023    
024        /**
025         * In this test, we just use simple TextWebSocketFrame to send publish/subscribe requests.
026         */
027        @Test
028        public void subscribeToOurselves() {
029            try {
030                WebSocketClientFactory clientFactory = new NettyWebSocketClientFactory();
031                WebSocketClient client = clientFactory.newClient(getAddressOfNode("1"),
032                        new WebSocketActionListener() {
033                            @Override
034                            public void onConnect(WebSocketClient client) {
035                                try {
036                                    logger.info("sending subscribe command");
037                                    client.send(new TextWebSocketFrame("{\"type\":\"subscribe\",\"data\":{\"subscriber\":\"mypubsubdemo\",\"topic\":\"demo\"}}"));
038                                    Thread.sleep(500);
039                                    logger.info("sending publish command (to ourselves)");
040                                    client.send(new TextWebSocketFrame("{\"type\":\"publish\",\"data\":{\"message\":\"Hello World\",\"topic\":\"demo\"}}"));
041                                } catch (Exception e) {
042                                    logger.error(e.getMessage(), e);
043                                }
044                            }
045    
046                            @Override
047                            public void onDisconnect(WebSocketClient client) {
048                                logger.info("web socket disconnected");
049                            }
050    
051                            @Override
052                            public void onMessage(WebSocketClient client, WebSocketFrame frame) {
053                                logger.info("frame received: {}", frame);
054                            }
055    
056                            @Override
057                            public void onError(Throwable t) {
058                                logger.error(t.getMessage(), t);
059                            }
060                        });
061                client.connect().await(1000, TimeUnit.MILLISECONDS);            
062                Thread.sleep(1000);
063                client.send(new CloseWebSocketFrame());
064                Thread.sleep(1000);
065                client.disconnect();
066            } catch (Exception e) {
067                logger.error(e.getMessage(), e);
068            }
069        }
070    
071    }
