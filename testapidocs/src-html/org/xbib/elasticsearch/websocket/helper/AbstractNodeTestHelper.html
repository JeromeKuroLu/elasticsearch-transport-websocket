002package org.xbib.elasticsearch.websocket.helper;
003
004import org.elasticsearch.action.admin.cluster.node.info.NodesInfoRequest;
005import org.elasticsearch.action.admin.cluster.node.info.NodesInfoResponse;
006import org.elasticsearch.action.admin.indices.create.CreateIndexRequest;
007import org.elasticsearch.action.admin.indices.delete.DeleteIndexRequest;
008import org.elasticsearch.client.Client;
009import org.elasticsearch.common.logging.ESLogger;
010import org.elasticsearch.common.logging.ESLoggerFactory;
011import org.elasticsearch.common.network.NetworkUtils;
012import org.elasticsearch.common.settings.ImmutableSettings;
013import org.elasticsearch.common.settings.Settings;
014import org.elasticsearch.common.transport.InetSocketTransportAddress;
015import org.elasticsearch.indices.IndexMissingException;
016import org.elasticsearch.node.Node;
017import org.junit.After;
018import org.junit.Before;
019
020import java.net.URI;
021import java.util.Map;
022
023import static org.elasticsearch.common.collect.Maps.newHashMap;
024import static org.elasticsearch.common.settings.ImmutableSettings.Builder.EMPTY_SETTINGS;
025import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
026import static org.elasticsearch.node.NodeBuilder.nodeBuilder;
027
028public abstract class AbstractNodeTestHelper {
029
030    private final static ESLogger logger = ESLoggerFactory.getLogger("test");
031
032    protected final String CLUSTER = "test-cluster-" + NetworkUtils.getLocalAddress().getHostName();
033
034    protected final String INDEX = "test-" + NetworkUtils.getLocalAddress().getHostName().toLowerCase();
035
036    protected Settings defaultSettings = ImmutableSettings
037            .settingsBuilder()
038            .put("cluster.name", CLUSTER)
039                    // default for queue_size for bulk thread pool is only 50 since 0.90.7
040                    // enlarge queue for this unit test because we are on a single machine with SSD...
041            .put("threadpool.bulk.queue_size", 200)
042            .build();
043
044    private Map<String, Node> nodes = newHashMap();
045
046    private Map<String, Client> clients = newHashMap();
047
048    private Map<String, InetSocketTransportAddress> addresses = newHashMap();
049
050    protected URI getAddressOfNode(String n) {
051        InetSocketTransportAddress address = addresses.get(n);
052        return URI.create("ws://"+address.address().getHostName()+":"
053                + (address.address().getPort() + 100) + "/websocket");
054    }
055
056    @Before
057    public void setUp() throws Exception {
058        startNode("1");
059        // find node address
060        NodesInfoRequest nodesInfoRequest = new NodesInfoRequest().transport(true);
061        NodesInfoResponse response = client("1").admin().cluster().nodesInfo(nodesInfoRequest).actionGet();
062        InetSocketTransportAddress address = (InetSocketTransportAddress)response.iterator().next()
063                        .getTransport().getAddress().publishAddress();
064        logger.info("address = {}", address);
065        addresses.put("1", address);
066        logger.info("creating index {}", INDEX);
067        client("1").admin().indices().create(new CreateIndexRequest(INDEX)).actionGet();
068        logger.info("index {} created", INDEX);
069    }
070
071    @After
072    public void deleteIndices() {
073        logger.info("deleting index {}", INDEX);
074        try {
075            client("1").admin().indices().delete(new DeleteIndexRequest().indices(INDEX)).actionGet();
076        } catch (IndexMissingException e) {
077            // ignore
078        }
079        logger.info("index {} deleted", INDEX);
080        closeAllNodes();
081    }
082
083    public Node startNode(String id) {
084        return buildNode(id).start();
085    }
086
087    public Node buildNode(String id) {
088        return buildNode(id, EMPTY_SETTINGS);
089    }
090
091    public Node buildNode(String id, Settings settings) {
092        String settingsSource = getClass().getName().replace('.', '/') + ".yml";
093        Settings finalSettings = settingsBuilder()
094                .loadFromClasspath(settingsSource)
095                .put(defaultSettings)
096                .put(settings)
097                .put("name", id)
098                .build();
099        if (finalSettings.get("gateway.type") == null) {
100            finalSettings = settingsBuilder().put(finalSettings).put("gateway.type", "none").build();
101        }
102        if (finalSettings.get("cluster.routing.schedule") != null) {
103            finalSettings = settingsBuilder().put(finalSettings).put("cluster.routing.schedule", "50ms").build();
104        }
105        Node node = nodeBuilder().settings(finalSettings).build();
106        Client client = node.client();
107        nodes.put(id, node);
108        clients.put(id, client);
109        return node;
110    }
111
112    public void stopNode(String id) {
113        Client client = clients.remove(id);
114        if (client != null) {
115            client.close();
116        }
117        Node node = nodes.remove(id);
118        if (node != null) {
119            node.close();
120        }
121    }
122
123    public Client client(String id) {
124        return clients.get(id);
125    }
126
127    public void closeAllNodes() {
128        for (Client client : clients.values()) {
129            client.close();
130        }
131        clients.clear();
132        for (Node node : nodes.values()) {
133            if (node != null) {
134                node.close();
135            }
136        }
137        nodes.clear();
138    }
139
140}
