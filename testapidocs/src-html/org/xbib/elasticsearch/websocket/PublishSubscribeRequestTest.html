023    public class PublishSubscribeRequestTest extends AbstractNodeTestHelper {
024        
025        private final static ESLogger logger = ESLoggerFactory.getLogger("test");    
026        
027        public void testOneClient() {
028            try {
029                final String subscriberId = "oneclient";
030                final String topic = "oneclienttest";
031                final WebSocketClientFactory clientFactory = new NettyWebSocketClientFactory();
032                
033                final WebSocketClientRequest subscribe = clientFactory.newRequest()
034                        .type("subscribe").data(jsonBuilder().startObject()
035                        .field("subscriber", "singleclient")
036                        .field("topic", topic).endObject());
037                
038                final WebSocketClientRequest publish = clientFactory.newRequest()
039                        .type("publish").data(jsonBuilder().startObject()
040                        .field("message", "Hello World")
041                        .field("topic", topic).endObject());
042                
043                WebSocketClient client = clientFactory.newClient(getAddressOfNode("1"),
044                        new WebSocketActionListener.Adapter() {
045                            @Override
046                            public void onConnect(WebSocketClient client) throws IOException {
047                                    logger.info("sending subscribe command, channel = {}", client.channel());
048                                    subscribe.send(client);
049                                    logger.info("sending publish command (to ourselves), channel = {}", client.channel());
050                                    publish.send(client);
051                            }
052    
053                            @Override
054                            public void onMessage(WebSocketClient client, WebSocketFrame frame) {
055                                logger.info("frame received: " + frame);
056                            }
057                        });
058                client.connect().await(1000, TimeUnit.MILLISECONDS);            
059                Thread.sleep(1000);
060                client.send(new CloseWebSocketFrame());
061                Thread.sleep(1000);
062                client.disconnect();
063                
064                clientFactory.shutdown();
065                
066            } catch (Exception e) {
067                logger.error(e.getMessage(), e);
068            }        
069        }
070    
071        @Test
072        public void testTwoClients() {
073            try {
074                final String subscriberId = "twoclients";
075                final String topic = "twoclienttest";
076                final WebSocketClientFactory clientFactory = new NettyWebSocketClientFactory();
077    
078                final WebSocketClientRequest subscribe = clientFactory.newRequest()
079                        .type("subscribe").data(jsonBuilder().startObject()
080                        .field("subscriber", "doubleclient")
081                        .field("topic", topic).endObject());
082                
083                final WebSocketClientRequest publish = clientFactory.newRequest()
084                        .type("publish").data(jsonBuilder().startObject()
085                        .field("message", "Hi there, I'm another client")
086                        .field("topic", topic).endObject());
087                
088                // open first client
089                WebSocketClient subscribingClient = clientFactory.newClient(getAddressOfNode("1"),
090                        new WebSocketActionListener.Adapter() {
091                            @Override
092                            public void onConnect(WebSocketClient client) {
093                                try {
094                                    logger.info("sending subscribe command, channel {}", client.channel());
095                                    subscribe.send(client);
096                                } catch (Exception e) {
097                                }
098                            }
099    
100                            @Override
101                            public void onMessage(WebSocketClient client, WebSocketFrame frame) {
102                                logger.info("frame {} received for subscribing client {}", frame, client.channel());
103                            }
104    
105                        });
106                
107                // open two client
108                WebSocketClient publishingClient = clientFactory.newClient(getAddressOfNode("1"),
109                        new WebSocketActionListener.Adapter() {
110                            @Override
111                            public void onConnect(WebSocketClient client) {
112                                try {
113                                    logger.info("sending publish command, channel = {}", client.channel());
114                                    publish.send(client);
115                                } catch (Exception e) {
116                                }
117                            }
118    
119                            @Override
120                            public void onMessage(WebSocketClient client, WebSocketFrame frame) {
121                                logger.info("frame {} received for publishing client {}", frame, client.channel());
122                            }
123    
124                        });
125                
126                // connect both clients to node
127                subscribingClient.connect().await(1000, TimeUnit.MILLISECONDS);            
128                publishingClient.connect().await(1000, TimeUnit.MILLISECONDS);            
129                
130                // wait for publish/subscribe
131                Thread.sleep(1000);
132                
133                // close first client
134                publishingClient.send(new CloseWebSocketFrame());
135                publishingClient.disconnect();
136    
137                // close second client
138                subscribingClient.send(new CloseWebSocketFrame());
139                subscribingClient.disconnect();
140                
141                Thread.sleep(1000);
142                
143                clientFactory.shutdown();
144                
145            } catch (Exception e) {
146                logger.error(e.getMessage(), e);
147            }        
148        }
