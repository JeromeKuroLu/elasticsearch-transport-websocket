063        super(settings);

065        this.transport = transport;
066        this.restController = restController;
067        this.httpPatchRestController = httpPatchRestController;
068        this.interActiveController = interActiveController;
069        this.nodeService = nodeService;
070        // the hook into nodeService can not be used here because of class cast
071        //nodeService.setHttpServer(this);

073        this.disableSites = componentSettings.getAsBoolean("disable_sites", false);
074
075        transport.httpServerAdapter(new Dispatcher(this));
076
077        transport.webSocketServerAdapter(new Interactor(this));
078
079    }

144    public void internalDispatchRequest(final HttpRequest request, final HttpChannel channel) {
145        if (request.rawPath().startsWith("/_plugin/")) {
146            RestFilterChain filterChain = restController.filterChain(pluginSiteFilter);
147            filterChain.continueProcessing(request, channel);
148            return;
149        }
150        restController.dispatchRequest(request, channel);
151        httpPatchRestController.dispatchRequest(request, channel);
152    }
153
154    public void internalPresence(Presence presence, String topic, Channel channel) {
155        interActiveController.presence(presence, topic, channel);
156    }
157
158    public void internalFrame(WebSocketServerHandshaker handshaker, WebSocketFrame frame, ChannelHandlerContext context) {
159        interActiveController.frame(handshaker, frame, context);
160    }

163    class PluginSiteFilter extends RestFilter {
164
165        @Override
166        public void process(RestRequest request, RestChannel channel, RestFilterChain filterChain) {
167            handlePluginSite((HttpRequest) request, (HttpChannel) channel);
168        }
169    }
170
171    void handlePluginSite(HttpRequest request, HttpChannel channel) {
172        if (disableSites) {
173            channel.sendResponse(new BytesRestResponse(FORBIDDEN));
174            return;
175        }
176        if (request.method().name().equals("OPTIONS")) {
177            // when we have OPTIONS request, simply send OK by default (with the Access Control Origin header which gets automatically added)
178            channel.sendResponse(new BytesRestResponse(OK));
179            return;
180        }
181        if (request.method().name().equals("GET")) {
182            channel.sendResponse(new BytesRestResponse(FORBIDDEN));
183            return;
184        }
185        // TODO for a "/_plugin" endpoint, we should have a page that lists all the plugins?
186
187        String path = request.rawPath().substring("/_plugin/".length());
188        int i1 = path.indexOf('/');
189        String pluginName;
190        String sitePath;
191        if (i1 == -1) {
192            pluginName = path;
193            sitePath = null;
194            // If a trailing / is missing, we redirect to the right page #2654
195            channel.sendResponse(new BytesRestResponse(RestStatus.MOVED_PERMANENTLY, "text/html", "<head><meta http-equiv=\"refresh\" content=\"0; URL=" + request.rawPath() + "/\"></head>"));
196            return;
197        } else {
198            pluginName = path.substring(0, i1);
199            sitePath = path.substring(i1 + 1);
200        }
201
202        if (sitePath.length() == 0) {
203            sitePath = "/index.html";
204        }
205
206        // Convert file separators.
207        sitePath = sitePath.replace('/', File.separatorChar);
208
209        // this is a plugin provided site, serve it as static files from the plugin location
210        File siteFile = new File(new File(environment.pluginsFile(), pluginName), "_site");
211        File file = new File(siteFile, sitePath);
212        if (!file.exists() || file.isHidden()) {
213            channel.sendResponse(new BytesRestResponse(NOT_FOUND));
214            return;
215        }
216        if (!file.isFile()) {
217            // If it's not a dir, we send a 403
218            if (!file.isDirectory()) {
219                channel.sendResponse(new BytesRestResponse(FORBIDDEN));
220                return;
221            }
222            // We don't serve dir but if index.html exists in dir we should serve it
223            file = new File(file, "index.html");
224            if (!file.exists() || file.isHidden() || !file.isFile()) {
225                channel.sendResponse(new BytesRestResponse(FORBIDDEN));
226                return;
227            }
228        }
229        if (!file.getAbsolutePath().startsWith(siteFile.getAbsolutePath())) {
230            channel.sendResponse(new BytesRestResponse(FORBIDDEN));
231            return;
232        }
233        try {
234            byte[] data = Streams.copyToByteArray(file);
235            channel.sendResponse(new BytesRestResponse(OK, guessMimeType(sitePath), data));
236        } catch (IOException e) {
237            channel.sendResponse(new BytesRestResponse(INTERNAL_SERVER_ERROR));
238        }
239    }

242    // TODO: Don't respond with a mime type that violates the request's Accept header
243    private String guessMimeType(String path) {
244        int lastDot = path.lastIndexOf('.');
245        if (lastDot == -1) {
246            return "";
247        }
248        String extension = path.substring(lastDot + 1).toLowerCase(Locale.ROOT);
249        String mimeType = DEFAULT_MIME_TYPES.get(extension);
250        if (mimeType == null) {
251            return "";
252        }
253        return mimeType;
254    }
