036    private static final ChannelBuffer END_JSONP;
037
038    static {
039        BytesRef U_END_JSONP = new BytesRef();
040        UnicodeUtil.UTF16toUTF8(");", 0, ");".length(), U_END_JSONP);
041        END_JSONP = ChannelBuffers.wrappedBuffer(U_END_JSONP.bytes, U_END_JSONP.offset, U_END_JSONP.length);
042    }

044    private final NettyWebSocketServerTransport transport;
045
046    private final Channel channel;
047
048    private final org.jboss.netty.handler.codec.http.HttpRequest nettyRequest;

050
051    public NettyHttpChannel(NettyWebSocketServerTransport transport, Channel channel,
052                            NettyHttpRequest request) {
053        super(request);
054        this.transport = transport;
055        this.channel = channel;
056        this.nettyRequest = request.request();
057    }

059    @Override
060    public org.elasticsearch.common.io.stream.BytesStreamOutput newBytesOutput() {
061        throw new UnsupportedOperationException();
062    }
063
064    public BytesStreamOutput newXbibBytesOutput() {
065        return new ReleasableBytesStreamOutput(transport.bigArrays);
066    }

068    @Override
069    public void sendResponse(RestResponse response) {
070        // Decide whether to close the connection or not.
071        boolean http10 = nettyRequest.getProtocolVersion().equals(HttpVersion.HTTP_1_0);
072        boolean close =
073                HttpHeaders.Values.CLOSE.equalsIgnoreCase(nettyRequest.headers().get(HttpHeaders.Names.CONNECTION)) ||
074                        (http10 && !HttpHeaders.Values.KEEP_ALIVE.equalsIgnoreCase(nettyRequest.headers().get(HttpHeaders.Names.CONNECTION)));
075
076        // Build the response object.
077        HttpResponseStatus status = getStatus(response.status());
078        org.jboss.netty.handler.codec.http.HttpResponse resp;
079        if (http10) {
080            resp = new DefaultHttpResponse(HttpVersion.HTTP_1_0, status);
081            if (!close) {
082                resp.headers().add(HttpHeaders.Names.CONNECTION, "Keep-Alive");
083            }
084        } else {
085            resp = new DefaultHttpResponse(HttpVersion.HTTP_1_1, status);
086        }
087        if (RestUtils.isBrowser(nettyRequest.headers().get(HttpHeaders.Names.USER_AGENT))) {
088            // add support for cross origin
089            resp.headers().add("Access-Control-Allow-Origin", transport.settings().get("http.cors.allow-origin", "*"));
090            if (nettyRequest.getMethod() == HttpMethod.OPTIONS) {
091                // Allow Ajax requests based on the CORS "preflight" request
092                resp.headers().add("Access-Control-Max-Age", transport.settings().getAsInt("http.cors.max-age", 1728000));
093                resp.headers().add("Access-Control-Allow-Methods", transport.settings().get("http.cors.allow-methods", "OPTIONS, HEAD, GET, POST, PUT, DELETE"));
094                resp.headers().add("Access-Control-Allow-Headers", transport.settings().get("http.cors.allow-headers", "X-Requested-With, Content-Type, Content-Length"));
095            }
096        }
097
098        String opaque = nettyRequest.headers().get("X-Opaque-Id");
099        if (opaque != null) {
100            resp.headers().add("X-Opaque-Id", opaque);
101        }
102
103        // Add all custom headers
104        Map<String, List<String>> customHeaders = response.getHeaders();
105        if (customHeaders != null) {
106            for (Map.Entry<String, List<String>> headerEntry : customHeaders.entrySet()) {
107                for (String headerValue : headerEntry.getValue()) {
108                    resp.headers().add(headerEntry.getKey(), headerValue);
109                }
110            }
111        }
112
113        ChannelBuffer buffer;
114        org.elasticsearch.common.bytes.BytesReference content = response.content();
115        boolean addedReleaseListener = false;
116        try {
117            buffer = response.contentThreadSafe() ?
118                    ChannelBuffers.wrappedBuffer(content.toBytes(), 0, content.length()) :
119                    ChannelBuffers.copiedBuffer(content.toBytes(), 0, content.length());
120
121            // handle JSONP
122            String callback = request.param("callback");
123            if (callback != null) {
124                final BytesRef callbackBytes = new BytesRef(callback.length() * 4 + 1);
125                UnicodeUtil.UTF16toUTF8(callback, 0, callback.length(), callbackBytes);
126                callbackBytes.bytes[callbackBytes.length] = '(';
127                callbackBytes.length++;
128                buffer = ChannelBuffers.wrappedBuffer(
129                        ChannelBuffers.wrappedBuffer(callbackBytes.bytes, callbackBytes.offset, callbackBytes.length),
130                        buffer,
131                        ChannelBuffers.wrappedBuffer(END_JSONP)
132                );
133            }
134            resp.setContent(buffer);
135            resp.headers().add(HttpHeaders.Names.CONTENT_TYPE, response.contentType());
136            resp.headers().add(HttpHeaders.Names.CONTENT_LENGTH, String.valueOf(buffer.readableBytes()));
137
138            if (transport.resetCookies) {
139                String cookieString = nettyRequest.headers().get(HttpHeaders.Names.COOKIE);
140                if (cookieString != null) {
141                    CookieDecoder cookieDecoder = new CookieDecoder();
142                    Set<Cookie> cookies = cookieDecoder.decode(cookieString);
143                    if (!cookies.isEmpty()) {
144                        // Reset the cookies if necessary.
145                        CookieEncoder cookieEncoder = new CookieEncoder(true);
146                        for (Cookie cookie : cookies) {
147                            cookieEncoder.addCookie(cookie);
148                        }
149                        resp.headers().add(HttpHeaders.Names.SET_COOKIE, cookieEncoder.encode());
150                    }
151                }
152            }
153
154            ChannelFuture future = channel.write(resp);
155            if (response.contentThreadSafe() && content instanceof Releasable) {
156                future.addListener(new ReleaseChannelFutureListener((Releasable) content));
157                addedReleaseListener = true;
158            }
159            if (close) {
160                future.addListener(ChannelFutureListener.CLOSE);
161            }
162        } finally {
163            if (!addedReleaseListener && content instanceof Releasable) {
164                ((Releasable) content).close();
165            }
166        }
167    }

169    private HttpResponseStatus getStatus(RestStatus status) {
170        switch (status) {
171            case CONTINUE:
172                return HttpResponseStatus.CONTINUE;
173            case SWITCHING_PROTOCOLS:
174                return HttpResponseStatus.SWITCHING_PROTOCOLS;
175            case OK:
176                return HttpResponseStatus.OK;
177            case CREATED:
178                return HttpResponseStatus.CREATED;
179            case ACCEPTED:
180                return HttpResponseStatus.ACCEPTED;
181            case NON_AUTHORITATIVE_INFORMATION:
182                return HttpResponseStatus.NON_AUTHORITATIVE_INFORMATION;
183            case NO_CONTENT:
184                return HttpResponseStatus.NO_CONTENT;
185            case RESET_CONTENT:
186                return HttpResponseStatus.RESET_CONTENT;
187            case PARTIAL_CONTENT:
188                return HttpResponseStatus.PARTIAL_CONTENT;
189            case MULTI_STATUS:
190                // no status for this??
191                return HttpResponseStatus.INTERNAL_SERVER_ERROR;
192            case MULTIPLE_CHOICES:
193                return HttpResponseStatus.MULTIPLE_CHOICES;
194            case MOVED_PERMANENTLY:
195                return HttpResponseStatus.MOVED_PERMANENTLY;
196            case FOUND:
197                return HttpResponseStatus.FOUND;
198            case SEE_OTHER:
199                return HttpResponseStatus.SEE_OTHER;
200            case NOT_MODIFIED:
201                return HttpResponseStatus.NOT_MODIFIED;
202            case USE_PROXY:
203                return HttpResponseStatus.USE_PROXY;
204            case TEMPORARY_REDIRECT:
205                return HttpResponseStatus.TEMPORARY_REDIRECT;
206            case BAD_REQUEST:
207                return HttpResponseStatus.BAD_REQUEST;
208            case UNAUTHORIZED:
209                return HttpResponseStatus.UNAUTHORIZED;
210            case PAYMENT_REQUIRED:
211                return HttpResponseStatus.PAYMENT_REQUIRED;
212            case FORBIDDEN:
213                return HttpResponseStatus.FORBIDDEN;
214            case NOT_FOUND:
215                return HttpResponseStatus.NOT_FOUND;
216            case METHOD_NOT_ALLOWED:
217                return HttpResponseStatus.METHOD_NOT_ALLOWED;
218            case NOT_ACCEPTABLE:
219                return HttpResponseStatus.NOT_ACCEPTABLE;
220            case PROXY_AUTHENTICATION:
221                return HttpResponseStatus.PROXY_AUTHENTICATION_REQUIRED;
222            case REQUEST_TIMEOUT:
223                return HttpResponseStatus.REQUEST_TIMEOUT;
224            case CONFLICT:
225                return HttpResponseStatus.CONFLICT;
226            case GONE:
227                return HttpResponseStatus.GONE;
228            case LENGTH_REQUIRED:
229                return HttpResponseStatus.LENGTH_REQUIRED;
230            case PRECONDITION_FAILED:
231                return HttpResponseStatus.PRECONDITION_FAILED;
232            case REQUEST_ENTITY_TOO_LARGE:
233                return HttpResponseStatus.REQUEST_ENTITY_TOO_LARGE;
234            case REQUEST_URI_TOO_LONG:
235                return HttpResponseStatus.REQUEST_URI_TOO_LONG;
236            case UNSUPPORTED_MEDIA_TYPE:
237                return HttpResponseStatus.UNSUPPORTED_MEDIA_TYPE;
238            case REQUESTED_RANGE_NOT_SATISFIED:
239                return HttpResponseStatus.REQUESTED_RANGE_NOT_SATISFIABLE;
240            case EXPECTATION_FAILED:
241                return HttpResponseStatus.EXPECTATION_FAILED;
242            case UNPROCESSABLE_ENTITY:
243                return HttpResponseStatus.BAD_REQUEST;
244            case LOCKED:
245                return HttpResponseStatus.BAD_REQUEST;
246            case FAILED_DEPENDENCY:
247                return HttpResponseStatus.BAD_REQUEST;
248            case INTERNAL_SERVER_ERROR:
249                return HttpResponseStatus.INTERNAL_SERVER_ERROR;
250            case NOT_IMPLEMENTED:
251                return HttpResponseStatus.NOT_IMPLEMENTED;
252            case BAD_GATEWAY:
253                return HttpResponseStatus.BAD_GATEWAY;
254            case SERVICE_UNAVAILABLE:
255                return HttpResponseStatus.SERVICE_UNAVAILABLE;
256            case GATEWAY_TIMEOUT:
257                return HttpResponseStatus.GATEWAY_TIMEOUT;
258            case HTTP_VERSION_NOT_SUPPORTED:
259                return HttpResponseStatus.HTTP_VERSION_NOT_SUPPORTED;
260            default:
261                return HttpResponseStatus.INTERNAL_SERVER_ERROR;
262        }
263    }
