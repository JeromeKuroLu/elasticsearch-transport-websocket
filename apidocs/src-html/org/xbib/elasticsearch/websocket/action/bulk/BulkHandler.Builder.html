036    /**
037     * The bulk handler is derived from the BulkProcessor, but
038     * offers explicit flushing and can receive requests from multiple threads
039     * into a global action queue.
040     * It supports websocket bulk actions and can write back response to an
041     * interactive channel.
042     * The bulk volume is not controlled.
043     * The default concurrency is 32, the number of actions in a bulk is 100.
044     */
045    public class BulkHandler extends BaseInteractiveHandler {
046    
047        private final BulkHandler.Listener listener;
048    
049        private final int concurrentRequests;
050    
051        private final int bulkActions;
052    
053        private final TimeValue flushInterval;
054    
055        private final Semaphore semaphore;
056    
057        private final ScheduledThreadPoolExecutor scheduler;
058    
059        private final ScheduledFuture scheduledFuture;
060    
061        private final AtomicLong executionIdGen = new AtomicLong();
062    
063        private final static Queue<ActionRequest> bulk = ConcurrentCollections.newQueue();
064    
065        private volatile boolean closed = false;
066    
067        @Override
068        public void handleRequest(final InteractiveRequest request, final InteractiveChannel channel) {
069            // will be overriden by bulk action
070        }

072        /**
073         * A listener for the execution.
074         */
075        public static interface Listener {
076    
077            /**
078             * Callback before the bulk is executed.
079             */
080            void beforeBulk(long executionId, BulkRequest request);
081    
082            /**
083             * Callback after a successful execution of bulk request.
084             */
085            void afterBulk(long executionId, BulkRequest request, BulkResponse response);
086    
087            /**
088             * Callback after a failed execution of bulk request.
089             */
090            void afterBulk(long executionId, BulkRequest request, Throwable failure);
091        }

123            /**
124             * Creates a builder of bulk processor with the client to use and the
125             * listener that will be used to be notified on the completion of bulk
126             * requests.
127             */
128            public Builder(Client client, BulkHandler.Listener listener) {
129                this.client = client;
130                this.listener = listener;
131            }
132    
133            /**
134             * Sets the number of concurrent requests allowed to be executed. A
135             * value of 0 means that only a single request will be allowed to be
136             * executed. A value of 1 means 1 concurrent request is allowed to be
137             * executed while accumulating new bulk requests. Defaults to
138             * <tt>1</tt>.
139             */
140            public BulkHandler.Builder setConcurrentRequests(int concurrentRequests) {
141                this.concurrentRequests = concurrentRequests;
142                return this;
143            }
144    
145            /**
146             * Sets when to flush a new bulk request based on the number of actions
147             * currently added. Defaults to <tt>1000</tt>. Can be set to <tt>-1</tt>
148             * to disable it.
149             */
150            public BulkHandler.Builder setBulkActions(int bulkActions) {
151                this.bulkActions = bulkActions;
152                return this;
153            }
154    
155            /**
156             * Sets a flush interval flushing *any* bulk actions pending if the
157             * interval passes. Defaults to not set.
158             * <p/>
159             * Note, {@link #setBulkActions(int)} can
160             * be set to <tt>-1</tt> with the flush interval set allowing for
161             * complete async processing of bulk actions.
162             */
163            public BulkHandler.Builder setFlushInterval(TimeValue flushInterval) {
164                this.flushInterval = flushInterval;
165                return this;
166            }

168            /**
169             * Builds a new bulk processor.
170             */
171            public BulkHandler build() {
172                return new BulkHandler(settings, client, listener, concurrentRequests, bulkActions, flushInterval);
173            }

176        public static BulkHandler.Builder builder(Client client, BulkHandler.Listener listener) {

178        }

180        public BulkHandler(Settings settings, Client client) {
181            super(settings, client);
182            this.listener = new ListenerAdapter();
183            this.concurrentRequests = 32;
184            this.bulkActions = 100;
185            this.semaphore = new Semaphore(concurrentRequests);
186            this.flushInterval = null;
187            this.scheduler = null;
188            this.scheduledFuture = null;
189        }
190    
191        BulkHandler(Settings settings, Client client, BulkHandler.Listener listener, int concurrentRequests, int bulkActions, @Nullable TimeValue flushInterval) {
192            super(settings, client);
193            this.listener = listener;
194            this.concurrentRequests = concurrentRequests;
195            this.bulkActions = bulkActions;
196            this.semaphore = new Semaphore(concurrentRequests);
197            this.flushInterval = flushInterval;
198            if (flushInterval != null) {
199                this.scheduler = (ScheduledThreadPoolExecutor) Executors.newScheduledThreadPool(1, EsExecutors.daemonThreadFactory(((InternalClient) client).settings(), "websocket_bulk_processor"));
200                this.scheduler.setExecuteExistingDelayedTasksAfterShutdownPolicy(false);
201                this.scheduler.setContinueExistingPeriodicTasksAfterShutdownPolicy(false);
202                this.scheduledFuture = this.scheduler.scheduleWithFixedDelay(new BulkHandler.Flush(), flushInterval.millis(), flushInterval.millis(), TimeUnit.MILLISECONDS);
203            } else {
204                this.scheduler = null;
205                this.scheduledFuture = null;
206            }
207        }
208    
209        public BulkHandler.Listener getListener() {
210            return listener;
211        }
212    
213        /**
214         * Flushes open bulk actions
215         */
216        public synchronized void flush() {
217            if (closed) {
218                return;
219            }
220            if (bulk.size() > 0) {
221                execute(null);
222            }
223        }
224    
225        /**
226         * Closes the processor. If flushing by time is enabled, then its shutdown.
227         * Any remaining bulk actions are flushed.
228         */
229        public synchronized void close() {
230            if (closed) {
231                return;
232            }
233            closed = true;
234            if (this.scheduledFuture != null) {
235                this.scheduledFuture.cancel(false);
236                this.scheduler.shutdown();
237            }
238            if (bulk.size() > 0) {
239                execute(null);
240            }
241        }

264        private synchronized void internalAdd(ActionRequest request) {
265            bulk.add(request);
266            executeIfNeeded();
267        }

269        private void executeIfNeeded() {
270            if (closed) {
271                throw new ElasticsearchIllegalStateException("bulk process already closed");
272            }
273            if (!isOverTheLimit()) {
274                return;
275            }
276            execute(null);
277        }

295        public BulkHandler add(ActionRequest request, InteractiveChannel channel) {
296            internalAdd(request, channel);
297            return this;
298        }
299    
300        private synchronized void internalAdd(ActionRequest request, InteractiveChannel channel) {
301            bulk.add(request);
302            executeIfNeeded(channel);
303        }
304    
305        private void executeIfNeeded(InteractiveChannel channel) {
306            if (closed) {
307                throw new ElasticsearchIllegalStateException("bulk process already closed");
308            }
309            if (!isOverTheLimit()) {
310                return;
311            }
312            execute(channel);
313        }
314    
315        // (currently) needs to be executed under a lock
316        private void execute(final InteractiveChannel channel) {
317            final BulkRequest bulkRequest = new BulkRequest().add(bulk);
318            bulk.clear();
319    
320            final long executionId = executionIdGen.incrementAndGet();
321    
322            if (concurrentRequests == 0) {
323                // execute in a blocking fashion...
324                try {
325                    listener.beforeBulk(executionId, bulkRequest);
326                    listener.afterBulk(executionId, bulkRequest, client.bulk(bulkRequest).actionGet());
327                } catch (Exception e) {
328                    listener.afterBulk(executionId, bulkRequest, e);
329                }
330            } else {
331                try {
332                    semaphore.acquire();
333                } catch (InterruptedException e) {
334                    listener.afterBulk(executionId, bulkRequest, e);
335                    return;
336                }
337                listener.beforeBulk(executionId, bulkRequest);
338                client.bulk(bulkRequest, new ActionListener<BulkResponse>() {
339                    @Override
340                    public void onResponse(BulkResponse response) {
341                        try {
342                            listener.afterBulk(executionId, bulkRequest, response);
343                            if (channel != null) {
344                                channel.sendResponse("bulkresponse", buildResponse(response));
345                            }
346                        } catch (IOException e) {
347                            logger.error("error while sending bulk response", e);
348                        } finally {
349                            semaphore.release();
350                        }
351                    }
352    
353                    @Override
354                    public void onFailure(Throwable t) {
355                        try {
356                            listener.afterBulk(executionId, bulkRequest, t);
357                            if (channel != null) {
358                                channel.sendResponse("bulkresponse", t);
359                            }
360                        } catch (IOException e) {
361                            logger.error("error while sending bulk response", e);
362                        } finally {
363                            semaphore.release();
364                        }
365                    }
366                });
367            }
368        }
369    
370        private boolean isOverTheLimit() {
371            if (bulkActions != -1 && bulk.size() > bulkActions) {
372                return true;
373            }
374            return false;
375        }
376    
377        class Flush implements Runnable {
378    
379            @Override
380            public void run() {
381                synchronized (BulkHandler.this) {
382                    if (closed) {
383                        return;
384                    }
385                    if (bulk.size() > 0) {
386                        execute(null);
387                    }
388                }
389            }
390        }

392        /**
393         * Taken from the REST bulk action.
394         * @param response the bulk response
395         * @return a content builder with the response
396         * @throws IOException 
397         */
398        private XContentBuilder buildResponse(BulkResponse response) throws IOException {
399            XContentBuilder builder = jsonBuilder();
400            builder.startObject();
401            builder.field(Fields.TOOK, response.getTookInMillis());
402            builder.startArray(Fields.ITEMS);
403            for (BulkItemResponse itemResponse : response) {
404                builder.startObject();
405                builder.startObject(itemResponse.getOpType());
406                builder.field(Fields._INDEX, itemResponse.getIndex());
407                builder.field(Fields._TYPE, itemResponse.getType());
408                builder.field(Fields._ID, itemResponse.getId());
409                long version = itemResponse.getVersion();
410                if (version != -1) {
411                    builder.field(Fields._VERSION, itemResponse.getVersion());
412                }
413                if (itemResponse.isFailed()) {
414                    builder.field(Fields.ERROR, itemResponse.getFailureMessage());
415                } else {
416                    builder.field(Fields.OK, true);
417                }
418                builder.endObject();
419                builder.endObject();
420            }
421            builder.endArray();
422            builder.endObject();
423            return builder;
424        }
