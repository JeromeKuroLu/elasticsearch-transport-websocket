001    
002    package org.xbib.elasticsearch.websocket.action.bulk;
003    
004    import org.elasticsearch.client.Client;
005    import org.elasticsearch.common.inject.Inject;
006    import org.elasticsearch.common.settings.Settings;
007    import org.xbib.elasticsearch.websocket.InteractiveChannel;
008    import org.xbib.elasticsearch.websocket.InteractiveController;
009    import org.xbib.elasticsearch.websocket.InteractiveRequest;
010    
011    /**
012     * Bulk flush action. This action forces bulk requests to get
013     * sent to the cluster.
014     */
015    public class BulkFlushAction extends BulkHandler {
016    
017        private final static String TYPE = "flush";
018    
019        @Inject
020        public BulkFlushAction(Settings settings,
021                Client client, InteractiveController controller) {
022            super(settings, client);
023            controller.registerHandler(TYPE, this);
024        }
025    
026        @Override
027        public void handleRequest(final InteractiveRequest request, final InteractiveChannel channel) {
028            this.flush();
029        }
030    }
