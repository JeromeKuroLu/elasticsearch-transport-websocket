002    package org.xbib.elasticsearch.websocket.action.pubsub;
003    
004    import java.io.IOException;
005    
006    import org.elasticsearch.ElasticsearchException;
007    import org.elasticsearch.action.delete.DeleteRequest;
008    import org.elasticsearch.action.get.GetResponse;
009    import org.elasticsearch.action.index.IndexRequest;
010    import org.elasticsearch.client.Client;
011    import org.elasticsearch.client.Requests;
012    import org.elasticsearch.common.component.AbstractLifecycleComponent;
013    import org.elasticsearch.common.inject.Inject;
014    import org.elasticsearch.common.logging.ESLogger;
015    import org.elasticsearch.common.logging.ESLoggerFactory;
016    import org.elasticsearch.common.settings.Settings;
017    import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
018    
019    import org.xbib.elasticsearch.websocket.InteractiveChannel;
020    import org.xbib.elasticsearch.websocket.action.bulk.BulkHandler;
021    
022    /*
023     * Checkpoint management for topics and subscribers.
024     */
025    public class Checkpointer extends AbstractLifecycleComponent<Checkpointer> {
026    
027        private final ESLogger logger = ESLoggerFactory.getLogger(Checkpointer.class.getSimpleName());
028    
029        private final String pubSubIndexName;
030    
031        private final Client client;
032    
033        private final static String TYPE = "checkpoint";
034    
035        private final BulkHandler bulkHandler;
036    
037        @Inject
038        public Checkpointer(Settings settings, Client client) {
039            super(settings);
040            this.client = client;
041            this.bulkHandler = new BulkHandler(settings, client);
042            this.pubSubIndexName = PubSubIndexName.Conf.indexName(settings);
043        }
044    
045        @Override
046        protected void doStart() throws ElasticsearchException {
047        }
048    
049        @Override
050        protected void doStop() throws ElasticsearchException {
051        }
052    
053        @Override
054        protected void doClose() throws ElasticsearchException {
055        }
056    
057        /**
058         * Checkpointing a topic or a subscriber. The current timestamp is written
059         * to the checkpoint index type. Note that bulk index requests are used by
060         * checkpointing and flushCheckpoint() needs to be called after all is done.
061         *
062         * @param id topic or subscriber
063         * @throws IOException
064         */
065        public void checkpoint(String id) throws IOException {
066            indexBulk(Requests.indexRequest(pubSubIndexName).type(TYPE).id(id)
067                    .source(jsonBuilder().startObject().field("timestamp", System.currentTimeMillis()).endObject()), null);
068        }
069    
070        public void flushCheckpoint() throws IOException {
071            flushBulk(null);
072        }
073    
074        public Long checkpointedAt(String id) throws IOException {
075            GetResponse response = client.prepareGet(pubSubIndexName, TYPE, id)
076                    .setFields("timestamp")
077                    .execute().actionGet();
078            boolean failed = !response.isExists();
079            if (failed) {
080                logger.warn("can't get checkpoint for {}", id);
081                return null;
082            } else {
083                return (Long) response.getFields().get("timestamp").getValue();
084            }
085        }
086    
087        /**
088         * Perform bulk indexing
089         *
090         * @param request the index request
091         * @param channel the interactive channel
092         * @throws IOException
093         */
094        public void indexBulk(IndexRequest request, InteractiveChannel channel) throws IOException {
095            bulkHandler.add(request, channel);
096        }
097    
098        /**
099         * Perform bulk delete
100         *
101         * @param request the delete request
102         * @param channel the interactive channel
103         * @throws IOException
104         */
105        public void deleteBulk(DeleteRequest request, InteractiveChannel channel) throws IOException {
106            bulkHandler.add(request, channel);
107        }
108    
109        /**
110         * Flush bulk
111         *
112         * @param channel the interactive channel
113         * @throws IOException
114         */
115        public void flushBulk(InteractiveChannel channel) throws IOException {
116            bulkHandler.flush();
117        }
118    }
