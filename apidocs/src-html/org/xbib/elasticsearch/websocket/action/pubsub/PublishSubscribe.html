002    package org.xbib.elasticsearch.websocket.action.pubsub;
003    
004    import org.elasticsearch.client.Client;
005    import org.elasticsearch.common.settings.Settings;
006    import org.elasticsearch.common.unit.TimeValue;
007    
008    import org.xbib.elasticsearch.websocket.BaseInteractiveHandler;
009    import org.xbib.elasticsearch.websocket.InteractiveController;
010    import org.xbib.elasticsearch.websocket.http.HttpServerTransport;
011    
012    /**
013     * Base class for Publish/Subscribe actions
014     */
015    public abstract class PublishSubscribe extends BaseInteractiveHandler {
016    
017        protected final String pubSubIndexName;
018    
019        protected final HttpServerTransport transport;
020    
021        protected final Checkpointer service;
022    
023        protected final TimeValue scrollTimeout = new TimeValue(60000);
024    
025        protected final int scrollSize = 100;
026        
027        public PublishSubscribe(Settings settings, 
028                Client client,
029                HttpServerTransport transport,
030                InteractiveController controller,
031                Checkpointer service) {
032            super(settings, client);
033            this.pubSubIndexName = PubSubIndexName.Conf.indexName(settings);
034            this.transport = transport;
035            this.service = service;
036        }
037        
038    }
