002    package org.xbib.elasticsearch.websocket.action.pubsub;
003    
004    import java.io.IOException;
005    import java.util.Map;
006    
007    import org.jboss.netty.channel.Channel;
008    
009    import org.elasticsearch.client.Client;
010    import org.elasticsearch.common.inject.Inject;
011    import org.elasticsearch.common.settings.Settings;
012    
013    import org.xbib.elasticsearch.websocket.BaseInteractiveHandler;
014    import org.xbib.elasticsearch.websocket.InteractiveChannel;
015    import org.xbib.elasticsearch.websocket.InteractiveController;
016    import org.xbib.elasticsearch.websocket.InteractiveRequest;
017    import org.xbib.elasticsearch.websocket.http.HttpServerTransport;
018    import org.xbib.elasticsearch.websocket.http.netty.NettyInteractiveResponse;
019    
020    /**
021     * Forwarding a message to a destination.
022     */
023    public class ForwardAction extends BaseInteractiveHandler {
024    
025        private final String TYPE = "forward";
026        private final HttpServerTransport transport;
027    
028        @Inject
029        public ForwardAction(Settings settings,
030                Client client,
031                HttpServerTransport transport,
032                InteractiveController controller) {
033            super(settings, client);
034            this.transport = transport;
035            controller.registerHandler(TYPE, this);
036        }
037    
038        @Override
039        public void handleRequest(final InteractiveRequest request, final InteractiveChannel channel) {
040            Map<String, Object> m = request.asMap();
041            Integer id = (Integer) m.get("channel");
042            Channel ch = transport.channel(id);
043            try {
044                if (ch != null) {
045                    ch.write(new NettyInteractiveResponse("message", (Map<String, Object>) m.get("message")).response());
046                    // don't send a success message back to the channel
047                } else {
048                    // delivery failed, channel not present
049                    channel.sendResponse(TYPE, new IOException("channel " + id + " gone"));
050                }
051            } catch (IOException ex) {
052                logger.error("error while delivering forward message {}: {}", m, ex);
053            }
054        }
055    }
