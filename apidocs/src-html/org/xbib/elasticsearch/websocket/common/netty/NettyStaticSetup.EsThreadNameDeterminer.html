002    package org.xbib.elasticsearch.websocket.common.netty;
003    
004    import org.jboss.netty.util.ThreadNameDeterminer;
005    import org.jboss.netty.util.ThreadRenamingRunnable;
006    
007    /**
008     */
009    public class NettyStaticSetup {
010    
011        private static EsThreadNameDeterminer ES_THREAD_NAME_DETERMINER = new EsThreadNameDeterminer();
012    
013        public static class EsThreadNameDeterminer implements ThreadNameDeterminer {
014            @Override
015            public String determineThreadName(String currentThreadName, String proposedThreadName) throws Exception {
016                // we control the thread name with a context, so use both
017                return currentThreadName + "{" + proposedThreadName + "}";
018            }
019        }
020    
021        static {
022            ThreadRenamingRunnable.setThreadNameDeterminer(ES_THREAD_NAME_DETERMINER);
023        }
024    
025        public static void setup() {
026    
027        }
028    }
