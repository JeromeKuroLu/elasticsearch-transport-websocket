002    package org.xbib.elasticsearch.websocket.common.netty;
003    
004    import org.jboss.netty.channel.Channel;
005    import org.jboss.netty.channel.ChannelEvent;
006    import org.jboss.netty.channel.ChannelFuture;
007    import org.jboss.netty.channel.ChannelFutureListener;
008    import org.jboss.netty.channel.ChannelHandler;
009    import org.jboss.netty.channel.ChannelHandlerContext;
010    import org.jboss.netty.channel.ChannelState;
011    import org.jboss.netty.channel.ChannelStateEvent;
012    import org.jboss.netty.channel.SimpleChannelUpstreamHandler;
013    
014    import java.util.Map;
015    
016    import org.elasticsearch.common.logging.ESLogger;
017    import org.elasticsearch.common.metrics.CounterMetric;
018    import org.elasticsearch.common.util.concurrent.ConcurrentCollections;
019    
020    /**
021     *
022     */
023    @ChannelHandler.Sharable
024    public class OpenChannelsHandler extends SimpleChannelUpstreamHandler {
025    
026        final Map<Integer,Channel> openChannels = ConcurrentCollections.newConcurrentMap();
027    
028        final CounterMetric openChannelsMetric = new CounterMetric();
029    
030        final CounterMetric totalChannelsMetric = new CounterMetric();
031    
032        final ESLogger logger;
033    
034        public OpenChannelsHandler(ESLogger logger) {
035            this.logger = logger;
036        }
037    
038        final ChannelFutureListener remover = new ChannelFutureListener() {
039            @Override
040            public void operationComplete(ChannelFuture future) throws Exception {
041                Channel channel = openChannels.remove(future.getChannel().getId());
042                if (channel != null) {
043                    openChannelsMetric.dec();
044                }
045                if (logger.isTraceEnabled()) {
046                    logger.trace("channel closed: {}", future.getChannel());
047                }
048            }
049        };
050    
051        @Override
052        public void handleUpstream(ChannelHandlerContext ctx, ChannelEvent e) throws Exception {
053            if (e instanceof ChannelStateEvent) {
054                ChannelStateEvent evt = (ChannelStateEvent) e;
055                // OPEN is also sent to when closing channel, but with FALSE on it to indicate it closes
056                if (evt.getState() == ChannelState.OPEN && Boolean.TRUE.equals(evt.getValue())) {
057                    if (logger.isTraceEnabled()) {
058                        logger.trace("channel opened: {}", ctx.getChannel());
059                    }
060                    Channel channel = openChannels.put(ctx.getChannel().getId(), ctx.getChannel());
061                    if (channel == null) {
062                        openChannelsMetric.inc();
063                        totalChannelsMetric.inc();
064                        ctx.getChannel().getCloseFuture().addListener(remover);
065                    }
066                }
067            }
068            ctx.sendUpstream(e);
069        }
070    
071        public Channel channel(Integer id) {
072            return openChannels.get(id);
073        }
074    
075        public long numberOfOpenChannels() {
076            return openChannelsMetric.count();
077        }
078    
079        public long totalChannels() {
080            return totalChannelsMetric.count();
081        }
082    
083        public void close() {
084            for (Channel channel : openChannels.values()) {
085                channel.close().awaitUninterruptibly();
086            }
087        }
088    }
