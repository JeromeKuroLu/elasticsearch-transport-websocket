001    
002    package org.xbib.elasticsearch.websocket.common.bytes;
003    
004    import java.io.IOException;
005    import java.io.OutputStream;
006    
007    import org.elasticsearch.common.io.stream.StreamInput;
008    import org.jboss.netty.buffer.ChannelBuffer;
009    
010    /**
011     * A reference to bytes for our Netty.
012     */
013    public interface BytesReference {
014    
015        /**
016         * Returns the byte at the specified index. Need to be between 0 and length.
017         */
018        byte get(int index);
019    
020        /**
021         * The length.
022         */
023        int length();
024    
025        /**
026         * Slice the bytes from the <tt>from</tt> index up to <tt>length</tt>.
027         */
028        BytesReference slice(int from, int length);
029    
030        /**
031         * A stream input of the bytes.
032         */
033        StreamInput streamInput();
034    
035        /**
036         * Writes the bytes directly to the output stream.
037         */
038        void writeTo(OutputStream os) throws IOException;
039    
040        /**
041         * Returns the bytes as a single byte array.
042         */
043        byte[] toBytes();
044    
045        /**
046         * Returns the bytes as a byte array, possibly sharing the underlying byte buffer.
047         */
048        BytesArray toBytesArray();
049    
050        /**
051         * Returns the bytes copied over as a byte array.
052         */
053        BytesArray copyBytesArray();
054    
055        /**
056         * Returns the bytes as a channel buffer.
057         */
058        ChannelBuffer toChannelBuffer();
059    
060        /**
061         * Is there an underlying byte array for this bytes reference.
062         */
063        boolean hasArray();
064    
065        /**
066         * The underlying byte array (if exists).
067         */
068        byte[] array();
069    
070        /**
071         * The offset into the underlying byte array.
072         */
073        int arrayOffset();
074    
075        /**
076         * Converts to a string based on utf8.
077         */
078        String toUtf8();
079    }
