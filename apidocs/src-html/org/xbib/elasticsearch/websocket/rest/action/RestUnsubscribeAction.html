002    package org.xbib.elasticsearch.websocket.rest.action;
003    
004    import java.io.IOException;
005    
006    import org.elasticsearch.action.ActionListener;
007    import org.elasticsearch.action.delete.DeleteResponse;
008    import org.elasticsearch.client.Client;
009    import org.elasticsearch.common.inject.Inject;
010    import org.elasticsearch.common.settings.Settings;
011    import org.elasticsearch.common.xcontent.XContentBuilder;
012    import org.elasticsearch.rest.BaseRestHandler;
013    import org.elasticsearch.rest.RestChannel;
014    import org.elasticsearch.rest.RestController;
015    import org.elasticsearch.rest.RestRequest;
016    import org.elasticsearch.rest.XContentRestResponse;
017    import org.elasticsearch.rest.XContentThrowableRestResponse;
018    
019    import static org.elasticsearch.rest.RestStatus.BAD_REQUEST;
020    import static org.elasticsearch.rest.RestStatus.OK;
021    import static org.elasticsearch.rest.action.support.RestXContentBuilder.restContentBuilder;
022    
023    import org.xbib.elasticsearch.websocket.action.pubsub.PubSubIndexName;
024    
025    /**
026     * Unsubscribe action for REST
027     * 
028     */
029    public class RestUnsubscribeAction extends BaseRestHandler {
030    
031        private final String pubSubIndexName;
032    
033        @Inject
034        public RestUnsubscribeAction(Settings settings, Client client, 
035                RestController restController) {
036            super(settings, client);
037            this.pubSubIndexName = PubSubIndexName.Conf.indexName(settings);
038            restController.registerHandler(RestRequest.Method.GET, "/_unsubscribe", this);
039            restController.registerHandler(RestRequest.Method.POST, "/_unsubscribe", this);
040        }
041    
042        @Override
043        public void handleRequest(final RestRequest request, final RestChannel channel) {
044            String subscriberId = request.hasParam("subscriber") ? request.param("subscriber") : null;
045            if (subscriberId == null) {
046                try {
047                    channel.sendResponse(new XContentThrowableRestResponse(request, new IllegalArgumentException("no subscriber")));
048                } catch (IOException e) {
049                    logger.error("error while sending failure response", e);
050                }
051                return;
052            }
053            try {
054                client.prepareDelete(pubSubIndexName, "subscribe", subscriberId)
055                        .execute(new ActionListener<DeleteResponse>() {
056                    @Override
057                    public void onResponse(DeleteResponse response) {
058                        try {
059                            XContentBuilder builder = restContentBuilder(request);
060                            builder.startObject().field("ok", true).field("id", response.getId()).endObject();
061                            channel.sendResponse(new XContentRestResponse(request, OK, builder));
062                        } catch (Exception e) {
063                            onFailure(e);
064                        }
065                    }
066    
067                    @Override
068                    public void onFailure(Throwable e) {
069                        try {
070                            logger.error("error while processing unsubscribe request", e);
071                            channel.sendResponse(new XContentThrowableRestResponse(request, e));
072                        } catch (IOException e1) {
073                            logger.error("error while sending error response", e1);
074                        }
075                    }
076                });
077            } catch (Exception e) {
078                try {
079                    XContentBuilder builder = restContentBuilder(request);
080                    channel.sendResponse(new XContentRestResponse(request, BAD_REQUEST, builder.startObject().field("error", e.getMessage()).endObject()));
081                } catch (IOException e1) {
082                    logger.error("exception while sending exception response", e1);
083                }
084            }
085        }   
086    
087    }
