002    package org.xbib.elasticsearch.websocket.client;
003    
004    import java.io.IOException;
005    import org.jboss.netty.handler.codec.http.websocketx.WebSocketFrame;
006    
007    /**
008     *  Listening to WebSocket actions.
009     * 
010     * @param <C>
011     * @param <F> 
012     */
013    public interface WebSocketActionListener<C extends WebSocketClient, F extends WebSocketFrame> {
014    
015        /**
016         * Called when the client is connected to the server
017         *
018         * @param client Current client used to connect
019         */
020        void onConnect(C client) throws IOException;
021    
022        /**
023         * Called when the client got disconnected from the server
024         *
025         * @param client Current client that was disconnected
026         */
027        void onDisconnect(C client) throws IOException;
028    
029        /**
030         * Called when a message arrives from the server
031         *
032         * @param client the connected client
033         * @param frame the data received from server
034         */
035        void onMessage(C client, F frame) throws IOException;
036    
037        /**
038         * Called when an unhandled errors occurs.
039         *
040         * @param t The causing error
041         */
042        void onError(Throwable t) throws IOException;
043    
044        class Adapter implements WebSocketActionListener {
045    
046            @Override
047            public void onConnect(WebSocketClient client) throws IOException {
048            }
049    
050            @Override
051            public void onDisconnect(WebSocketClient client) throws IOException {
052            }
053    
054            @Override
055            public void onMessage(WebSocketClient client, WebSocketFrame frame) throws IOException {
056            }
057    
058            @Override
059            public void onError(Throwable t) throws IOException {
060            }
061            
062        }
063        
064    }
