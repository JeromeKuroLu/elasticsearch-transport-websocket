002    package org.xbib.elasticsearch.websocket.transport.netty;
003    
004    import java.io.IOException;
005    import java.io.NotSerializableException;
006    
007    import org.elasticsearch.Version;
008    import org.elasticsearch.common.compress.CompressorFactory;
009    import org.elasticsearch.common.io.stream.HandlesStreamOutput;
010    import org.elasticsearch.common.io.stream.StreamOutput;
011    import org.elasticsearch.common.io.ThrowableObjectOutputStream;
012    import org.elasticsearch.transport.TransportResponse;
013    import org.elasticsearch.transport.NotSerializableTransportException;
014    import org.elasticsearch.transport.RemoteTransportException;
015    import org.elasticsearch.transport.TransportChannel;
016    import org.elasticsearch.transport.TransportResponseOptions;
017    import org.elasticsearch.transport.support.TransportStatus;
018    
019    import org.jboss.netty.buffer.ChannelBuffer;
020    import org.jboss.netty.channel.Channel;
021    
022    import org.xbib.elasticsearch.websocket.common.io.stream.BytesStreamOutput;
023    
024    /**
025     *
026     */
027    public class NettyTransportChannel implements TransportChannel {
028    
029        private final NettyTransport transport;
030        private final Version version;
031        private final String action;
032    
033        private final Channel channel;
034    
035        private final long requestId;
036    
037        public NettyTransportChannel(NettyTransport transport, String action, Channel channel, long requestId, Version version) {
038            this.version = version;
039            this.transport = transport;
040            this.action = action;
041            this.channel = channel;
042            this.requestId = requestId;
043        }
044    
045        @Override
046        public String action() {
047            return this.action;
048        }
049    
050        @Override
051        public void sendResponse(TransportResponse response) throws IOException {
052            sendResponse(response, TransportResponseOptions.EMPTY);
053        }
054    
055        @Override
056        public void sendResponse(TransportResponse response, TransportResponseOptions options) throws IOException {
057            if (transport.compress) {
058                options.withCompress(true);
059            }
060            byte status = 0;
061            status = TransportStatus.setResponse(status);
062    
063            BytesStreamOutput bStream = new BytesStreamOutput();
064            bStream.skip(org.elasticsearch.transport.netty.NettyHeader.HEADER_SIZE);
065            StreamOutput stream = bStream;
066            if (options.compress()) {
067                status = TransportStatus.setCompress(status);
068                stream = CompressorFactory.defaultCompressor().streamOutput(stream);
069            }
070            stream = new HandlesStreamOutput(stream);
071            stream.setVersion(version);
072            response.writeTo(stream);
073            stream.close();
074    
075            ChannelBuffer buffer = bStream.ourBytes().toChannelBuffer();
076            NettyHeader.writeHeader(buffer, requestId, status, version);
077            channel.write(buffer);
078    
079        }
080    
081        @Override
082        public void sendResponse(Throwable error) throws IOException {
083            BytesStreamOutput stream = new BytesStreamOutput();
084            try {
085                stream.skip(NettyHeader.HEADER_SIZE);
086                RemoteTransportException tx = new RemoteTransportException(transport.nodeName(), transport.wrapAddress(channel.getLocalAddress()), action, error);
087                ThrowableObjectOutputStream too = new ThrowableObjectOutputStream(stream);
088                too.writeObject(tx);
089                too.close();
090            } catch (NotSerializableException e) {
091                stream.reset();
092                stream.skip(org.elasticsearch.transport.netty.NettyHeader.HEADER_SIZE);
093                RemoteTransportException tx = new RemoteTransportException(transport.nodeName(), transport.wrapAddress(channel.getLocalAddress()), action, new NotSerializableTransportException(error));
094                ThrowableObjectOutputStream too = new ThrowableObjectOutputStream(stream);
095                too.writeObject(tx);
096                too.close();
097            }
098    
099            byte status = 0;
100            status = TransportStatus.setResponse(status);
101            status = TransportStatus.setError(status);
102    
103            ChannelBuffer buffer = stream.ourBytes().toChannelBuffer();
104            NettyHeader.writeHeader(buffer, requestId, status, version);
105            channel.write(buffer);
106        }
107    
108    }
