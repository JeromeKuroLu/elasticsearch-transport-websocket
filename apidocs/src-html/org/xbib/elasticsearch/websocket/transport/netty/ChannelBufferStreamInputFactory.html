002    package org.xbib.elasticsearch.websocket.transport.netty;
003    
004    import org.elasticsearch.common.io.stream.StreamInput;
005    import org.jboss.netty.buffer.ChannelBuffer;
006    
007    /**
008     */
009    public class ChannelBufferStreamInputFactory {
010    
011        public static StreamInput create(ChannelBuffer buffer) {
012            return new ChannelBufferStreamInput(buffer, buffer.readableBytes());
013        }
014    
015        public static StreamInput create(ChannelBuffer buffer, int size) {
016            return new ChannelBufferStreamInput(buffer, size);
017        }
018    }
