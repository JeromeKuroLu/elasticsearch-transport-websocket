002    package org.xbib.elasticsearch.websocket.transport.netty;
003    
004    import org.jboss.netty.buffer.ChannelBuffer;
005    import org.jboss.netty.channel.Channel;
006    import org.jboss.netty.channel.ChannelHandlerContext;
007    import org.jboss.netty.handler.codec.frame.TooLongFrameException;
008    import org.jboss.netty.handler.codec.frame.FrameDecoder;
009    
010    import java.io.StreamCorruptedException;
011    
012    import org.elasticsearch.common.unit.ByteSizeValue;
013    import org.elasticsearch.monitor.jvm.JvmInfo;
014    
015    /**
016     */
017    public class SizeHeaderFrameDecoder extends FrameDecoder {
018    
019        private static final long NINETY_PER_HEAP_SIZE = (long) (JvmInfo.jvmInfo().mem().heapMax().bytes() * 0.9);
020    
021        @Override
022        protected Object decode(ChannelHandlerContext ctx, Channel channel, ChannelBuffer buffer) throws Exception {
023            if (buffer.readableBytes() < 4) {
024                return null;
025            }
026    
027            int dataLen = buffer.getInt(buffer.readerIndex());
028            if (dataLen <= 0) {
029                throw new StreamCorruptedException("invalid data length: " + dataLen);
030            }
031            // safety against too large frames being sent
032            if (dataLen > NINETY_PER_HEAP_SIZE) {
033                throw new TooLongFrameException(
034                        "transport content length received [" + new ByteSizeValue(dataLen) + "] exceeded [" + new ByteSizeValue(NINETY_PER_HEAP_SIZE) + "]");
035            }
036    
037            if (buffer.readableBytes() < dataLen + 4) {
038                return null;
039            }
040            buffer.skipBytes(4);
041            return buffer;
042        }
043    }
