030    /**
031     * A handler (must be the last one!) that does size based frame decoding and forwards the actual message
032     * to the relevant action.
033     */
034    public class MessageChannelHandler extends SimpleChannelUpstreamHandler {
035    
036        private final ESLogger logger;
037    
038        private final ThreadPool threadPool;
039    
040        private final TransportServiceAdapter transportServiceAdapter;
041    
042        private final NettyTransport transport;
043    
044        public MessageChannelHandler(NettyTransport transport, ESLogger logger) {
045            this.threadPool = transport.threadPool();
046            this.transportServiceAdapter = transport.transportServiceAdapter();
047            this.transport = transport;
048            this.logger = logger;
049        }
050    
051        @Override
052        public void messageReceived(ChannelHandlerContext ctx, MessageEvent e) throws Exception {
053            Object m = e.getMessage();
054            if (!(m instanceof ChannelBuffer)) {
055                ctx.sendUpstream(e);
056                return;
057            }
058            ChannelBuffer buffer = (ChannelBuffer) m;
059            int size = buffer.getInt(buffer.readerIndex() - 4);
060            transportServiceAdapter.received(size + 4);
061    
062            int markedReaderIndex = buffer.readerIndex();
063            int expectedIndexReader = markedReaderIndex + size;
064    
065            // netty always copies a buffer, either in NioWorker in its read handler, where it copies to a fresh
066            // buffer, or in the cumlation buffer, which is cleaned each time
067            StreamInput streamIn = ChannelBufferStreamInputFactory.create(buffer, size);
068    
069            long requestId = buffer.readLong();
070            byte status = buffer.readByte();
071            Version version = Version.fromId(buffer.readInt());
072    
073            // !!! compression handling removed !!!
074            StreamInput wrappedStream = CachedStreamInput.cachedHandles(streamIn);
075    
076            if (TransportStatus.isRequest(status)) {
077                String action = handleRequest(ctx.getChannel(), wrappedStream, requestId, version);
078                if (buffer.readerIndex() != expectedIndexReader) {
079                    if (buffer.readerIndex() < expectedIndexReader) {
080                        logger.warn("Message not fully read (request) for [{}] and action [{}], resetting", requestId, action);
081                    } else {
082                        logger.warn("Message read past expected size (request) for [{}] and action [{}], resetting", requestId, action);
083                    }
084                    buffer.readerIndex(expectedIndexReader);
085                }
086            } else {
087                TransportResponseHandler handler = transportServiceAdapter.remove(requestId);
088                // ignore if its null, the adapter logs it
089                if (handler != null) {
090                    if (TransportStatus.isError(status)) {
091                        handlerResponseError(wrappedStream, handler);
092                    } else {
093                        handleResponse(wrappedStream, handler);
094                    }
095                } else {
096                    // if its null, skip those bytes
097                    buffer.readerIndex(markedReaderIndex + size);
098                }
099                if (buffer.readerIndex() != expectedIndexReader) {
100                    if (buffer.readerIndex() < expectedIndexReader) {
101                        logger.warn("Message not fully read (response) for [{}] handler {}, error [{}], resetting", requestId, handler, TransportStatus.isError(status));
102                    } else {
103                        logger.warn("Message read past expected size (response) for [{}] handler {}, error [{}], resetting", requestId, handler, TransportStatus.isError(status));
104                    }
105                    buffer.readerIndex(expectedIndexReader);
106                }
107            }
108            wrappedStream.close();
109        }
110    
111        private void handleResponse(StreamInput buffer, final TransportResponseHandler handler) {
112            final TransportResponse response = handler.newInstance();
113            try {
114                response.readFrom(buffer);
115            } catch (Throwable e) {
116                handleException(handler, new TransportSerializationException("Failed to deserialize response of type [" + response.getClass().getName() + "]", e));
117                return;
118            }
119            try {
120                if (handler.executor().equals(ThreadPool.Names.SAME)) {
121                    handler.handleResponse(response);
122                } else {
123                    threadPool.executor(handler.executor()).execute(new ResponseHandler(handler, response));
124                }
125            } catch (Throwable e) {
126                handleException(handler, new ResponseHandlerFailureTransportException(e));
127            }
128        }
129    
130        private void handlerResponseError(StreamInput buffer, final TransportResponseHandler handler) {
131            Throwable error;
132            try {
133                ThrowableObjectInputStream ois = new ThrowableObjectInputStream(buffer, transport.settings().getClassLoader());
134                error = (Throwable) ois.readObject();
135            } catch (Exception e) {
136                error = new TransportSerializationException("Failed to deserialize exception response from stream", e);
137            }
138            handleException(handler, error);
139        }
140    
141        private void handleException(final TransportResponseHandler handler, Throwable error) {
142            if (!(error instanceof RemoteTransportException)) {
143                error = new RemoteTransportException(error.getMessage(), error);
144            }
145            final RemoteTransportException rtx = (RemoteTransportException) error;
146            if (handler.executor().equals(ThreadPool.Names.SAME)) {
147                handler.handleException(rtx);
148            } else {
149                threadPool.executor(handler.executor()).execute(new Runnable() {
150                    @Override
151                    public void run() {
152                        try {
153                            handler.handleException(rtx);
154                        } catch (Exception e) {
155                            logger.error("Failed to handle exception response", e);
156                        }
157                    }
158                });
159            }
160        }
161    
162        private String handleRequest(Channel channel, StreamInput buffer, long requestId, Version version) throws IOException {
163            final String action = buffer.readString();
164    
165            final NettyTransportChannel transportChannel =
166                    new NettyTransportChannel(transport, action, channel, requestId, version);
167            try {
168                final TransportRequestHandler handler = transportServiceAdapter.handler(action);
169                if (handler == null) {
170                    throw new ActionNotFoundTransportException(action);
171                }
172                final TransportRequest request = handler.newInstance();
173                request.readFrom(buffer);
174                if (handler.executor().equals(ThreadPool.Names.SAME)) {
175                    //noinspection unchecked
176                    handler.messageReceived(request, transportChannel);
177                } else {
178                    threadPool.executor(handler.executor()).execute(new RequestHandler(handler, request, transportChannel, action));
179                }
180            } catch (Throwable e) {
181                try {
182                    transportChannel.sendResponse(e);
183                } catch (IOException e1) {
184                    logger.warn("Failed to send error message back to client for action [" + action + "]", e);
185                    logger.warn("Actual Exception", e1);
186                }
187            }
188            return action;
189        }
190    
191    
192        class ResponseHandler implements Runnable {
193    
194            private final TransportResponseHandler handler;
195            private final TransportResponse response;
196    
197            public ResponseHandler(TransportResponseHandler handler, TransportResponse response) {
198                this.handler = handler;
199                this.response = response;
200            }
201    
202            @SuppressWarnings({"unchecked"})
203            @Override
204            public void run() {
205                try {
206                    handler.handleResponse(response);
207                } catch (Exception e) {
208                    handleException(handler, new ResponseHandlerFailureTransportException(e));
209                }
210            }
211        }
212    
213        class RequestHandler implements Runnable {
214            private final TransportRequestHandler handler;
215            private final TransportRequest request;
216            private final NettyTransportChannel transportChannel;
217            private final String action;
218    
219            public RequestHandler(TransportRequestHandler handler,  TransportRequest request, NettyTransportChannel transportChannel, String action) {
220                this.handler = handler;
221                this.request = request;
222                this.transportChannel = transportChannel;
223                this.action = action;
224            }
225    
226            @SuppressWarnings({"unchecked"})
227            @Override
228            public void run() {
229                try {
230                    handler.messageReceived(request, transportChannel);
231                } catch (Throwable e) {
232                    if (transport.lifecycleState() == Lifecycle.State.STARTED) {
233                        // we can only send a response transport is started....
234                        try {
235                            transportChannel.sendResponse(e);
236                        } catch (IOException e1) {
237                            logger.warn("Failed to send error message back to client for action [" + action + "]", e1);
238                            logger.warn("Actual Exception", e);
239                        }
240                    }
241                }
242            }
243        }
