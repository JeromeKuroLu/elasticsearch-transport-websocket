002    package org.xbib.elasticsearch.websocket.transport.netty;
003    
004    import org.elasticsearch.Version;
005    import org.jboss.netty.buffer.ChannelBuffer;
006    
007    /**
008     */
009    public class NettyHeader {
010    
011        public static final int HEADER_SIZE = 2 + 4 + 8 + 1 + 4;
012    
013        public static void writeHeader(ChannelBuffer buffer, long requestId, byte status, Version version) {
014            int index = buffer.readerIndex();
015            buffer.setByte(index, 'E');
016            index += 1;
017            buffer.setByte(index, 'S');
018            index += 1;
019            // write the size, the size indicates the remaining message size, not including the size int
020            buffer.setInt(index, buffer.readableBytes() - 6);
021            index += 4;
022            buffer.setLong(index, requestId);
023            index += 8;
024            buffer.setByte(index, status);
025            index += 1;
026            buffer.setInt(index, version.id);
027        }
028    }
