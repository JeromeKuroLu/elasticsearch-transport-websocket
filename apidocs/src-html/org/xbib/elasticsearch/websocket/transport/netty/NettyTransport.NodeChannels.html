088    /**
089     * There are 4 types of connections per node, low/med/high/ping. Low if for batch oriented APIs (like recovery or
090     * batch) with high payload that will cause regular request. (like search or single index) to take
091     * longer. Med is for the typical search / single doc index. And High for things like cluster state. Ping is reserved for
092     * sending out ping requests to other nodes.
093     */

167        @Inject
168        public NettyTransport(Settings settings, ThreadPool threadPool, NetworkService networkService, Version version) {
169            super(settings);
170            this.threadPool = threadPool;
171            this.networkService = networkService;
172            this.version = version;
173    
174            if (settings.getAsBoolean("netty.epollBugWorkaround", false)) {
175                System.setProperty("org.jboss.netty.epollBugWorkaround", "true");
176            }
177    
178            this.workerCount = componentSettings.getAsInt("worker_count", EsExecutors.boundedNumberOfProcessors(settings) * 2);
179            this.bossCount = componentSettings.getAsInt("boss_count", 1);
180            this.blockingServer = settings.getAsBoolean("transport.tcp.blocking_server", settings.getAsBoolean(TCP_BLOCKING_SERVER, settings.getAsBoolean(TCP_BLOCKING, false)));
181            this.blockingClient = settings.getAsBoolean("transport.tcp.blocking_client", settings.getAsBoolean(TCP_BLOCKING_CLIENT, settings.getAsBoolean(TCP_BLOCKING, false)));
182            this.port = componentSettings.get("port", settings.get("transport.tcp.port", "9300-9400"));
183            this.bindHost = componentSettings.get("bind_host", settings.get("transport.bind_host", settings.get("transport.host")));
184            this.publishHost = componentSettings.get("publish_host", settings.get("transport.publish_host", settings.get("transport.host")));
185            this.compress = settings.getAsBoolean("transport.tcp.compress", false);
186            this.connectTimeout = componentSettings.getAsTime("connect_timeout", settings.getAsTime("transport.tcp.connect_timeout", settings.getAsTime(TCP_CONNECT_TIMEOUT, TCP_DEFAULT_CONNECT_TIMEOUT)));
187            this.tcpNoDelay = componentSettings.getAsBoolean("tcp_no_delay", settings.getAsBoolean(TCP_NO_DELAY, true));
188            this.tcpKeepAlive = componentSettings.getAsBoolean("tcp_keep_alive", settings.getAsBoolean(TCP_KEEP_ALIVE, true));
189            this.reuseAddress = componentSettings.getAsBoolean("reuse_address", settings.getAsBoolean(TCP_REUSE_ADDRESS, NetworkUtils.defaultReuseAddress()));
190            this.tcpSendBufferSize = componentSettings.getAsBytesSize("tcp_send_buffer_size", settings.getAsBytesSize(TCP_SEND_BUFFER_SIZE, TCP_DEFAULT_SEND_BUFFER_SIZE));
191            this.tcpReceiveBufferSize = componentSettings.getAsBytesSize("tcp_receive_buffer_size", settings.getAsBytesSize(TCP_RECEIVE_BUFFER_SIZE, TCP_DEFAULT_RECEIVE_BUFFER_SIZE));
192            this.connectionsPerNodeRecovery = componentSettings.getAsInt("connections_per_node.recovery", settings.getAsInt("transport.connections_per_node.recovery", 2));
193            this.connectionsPerNodeBulk = componentSettings.getAsInt("connections_per_node.bulk", settings.getAsInt("transport.connections_per_node.bulk", 3));
194            this.connectionsPerNodeReg = componentSettings.getAsInt("connections_per_node.reg", settings.getAsInt("transport.connections_per_node.reg", 6));
195            this.connectionsPerNodeState = componentSettings.getAsInt("connections_per_node.high", settings.getAsInt("transport.connections_per_node.state", 1));
196            this.connectionsPerNodePing = componentSettings.getAsInt("connections_per_node.ping", settings.getAsInt("transport.connections_per_node.ping", 1));
197    
198            // we want to have at least 1 for reg/state/ping
199            if (this.connectionsPerNodeReg == 0) {
200                throw new ElasticsearchIllegalArgumentException("can't set [connection_per_node.reg] to 0");
201            }
202            if (this.connectionsPerNodePing == 0) {
203                throw new ElasticsearchIllegalArgumentException("can't set [connection_per_node.ping] to 0");
204            }
205            if (this.connectionsPerNodeState == 0) {
206                throw new ElasticsearchIllegalArgumentException("can't set [connection_per_node.state] to 0");
207            }
208    
209            this.maxCumulationBufferCapacity = componentSettings.getAsBytesSize("max_cumulation_buffer_capacity", null);
210            this.maxCompositeBufferComponents = componentSettings.getAsInt("max_composite_buffer_components", -1);
211    
212            long defaultReceiverPredictor = 512 * 1024;
213    
214            // skip JVM info
215    
216            // See AdaptiveReceiveBufferSizePredictor#DEFAULT_XXX for default values in netty..., we can use higher ones for us, even fixed one
217            ByteSizeValue receivePredictorMin = componentSettings.getAsBytesSize("receive_predictor_min", componentSettings.getAsBytesSize("receive_predictor_size", new ByteSizeValue(defaultReceiverPredictor)));
218            ByteSizeValue receivePredictorMax = componentSettings.getAsBytesSize("receive_predictor_max", componentSettings.getAsBytesSize("receive_predictor_size", new ByteSizeValue(defaultReceiverPredictor)));
219            if (receivePredictorMax.bytes() == receivePredictorMin.bytes()) {
220                receiveBufferSizePredictorFactory = new FixedReceiveBufferSizePredictorFactory((int) receivePredictorMax.bytes());
221            } else {
222                receiveBufferSizePredictorFactory = new AdaptiveReceiveBufferSizePredictorFactory((int) receivePredictorMin.bytes(), (int) receivePredictorMin.bytes(), (int) receivePredictorMax.bytes());
223            }
224    
225            logger.debug("using worker_count[{}], port[{}], bind_host[{}], publish_host[{}], compress[{}], connect_timeout[{}], connections_per_node[{}/{}/{}/{}/{}], receive_predictor[{}->{}]",
226                    workerCount, port, bindHost, publishHost, compress, connectTimeout, connectionsPerNodeRecovery, connectionsPerNodeBulk, connectionsPerNodeReg, connectionsPerNodeState, connectionsPerNodePing, receivePredictorMin, receivePredictorMax);
227        }

246        @Override
247        protected void doStart() throws ElasticsearchException {
248            if (blockingClient) {
249                clientBootstrap = new ClientBootstrap(new OioClientSocketChannelFactory(Executors.newCachedThreadPool(daemonThreadFactory(settings, "transport_client_worker"))));
250            } else {
251                clientBootstrap = new ClientBootstrap(new NioClientSocketChannelFactory(
252                        Executors.newCachedThreadPool(daemonThreadFactory(settings, "transport_client_boss")),
253                        bossCount,
254                        new NioWorkerPool(Executors.newCachedThreadPool(daemonThreadFactory(settings, "transport_client_worker")), workerCount),
255                        new HashedWheelTimer(daemonThreadFactory(settings, "transport_client_timer"))));
256            }
257            ChannelPipelineFactory clientPipelineFactory = new ChannelPipelineFactory() {
258                @Override
259                public ChannelPipeline getPipeline() throws Exception {
260                    ChannelPipeline pipeline = Channels.pipeline();
261                    SizeHeaderFrameDecoder sizeHeader = new SizeHeaderFrameDecoder();
262                    if (maxCumulationBufferCapacity != null) {
263                        if (maxCumulationBufferCapacity.bytes() > Integer.MAX_VALUE) {
264                            sizeHeader.setMaxCumulationBufferCapacity(Integer.MAX_VALUE);
265                        } else {
266                            sizeHeader.setMaxCumulationBufferCapacity((int) maxCumulationBufferCapacity.bytes());
267                        }
268                    }
269                    if (maxCompositeBufferComponents != -1) {
270                        sizeHeader.setMaxCumulationBufferComponents(maxCompositeBufferComponents);
271                    }
272                    pipeline.addLast("size", sizeHeader);
273                    pipeline.addLast("dispatcher", new MessageChannelHandler(NettyTransport.this, logger));
274                    return pipeline;
275                }
276            };
277            clientBootstrap.setPipelineFactory(clientPipelineFactory);
278            clientBootstrap.setOption("connectTimeoutMillis", connectTimeout.millis());
279            if (tcpNoDelay != null) {
280                clientBootstrap.setOption("tcpNoDelay", tcpNoDelay);
281            }
282            if (tcpKeepAlive != null) {
283                clientBootstrap.setOption("keepAlive", tcpKeepAlive);
284            }
285            if (tcpSendBufferSize != null && tcpSendBufferSize.bytes() > 0) {
286                clientBootstrap.setOption("sendBufferSize", tcpSendBufferSize.bytes());
287            }
288            if (tcpReceiveBufferSize != null && tcpReceiveBufferSize.bytes() > 0) {
289                clientBootstrap.setOption("receiveBufferSize", tcpReceiveBufferSize.bytes());
290            }
291            clientBootstrap.setOption("receiveBufferSizePredictorFactory", receiveBufferSizePredictorFactory);
292            if (reuseAddress != null) {
293                clientBootstrap.setOption("reuseAddress", reuseAddress);
294            }
295    
296            if (!settings.getAsBoolean("network.server", true)) {
297                return;
298            }
299    
300            serverOpenChannels = new OpenChannelsHandler(logger);
301            if (blockingServer) {
302                serverBootstrap = new ServerBootstrap(new OioServerSocketChannelFactory(
303                        Executors.newCachedThreadPool(daemonThreadFactory(settings, "transport_server_boss")),
304                        Executors.newCachedThreadPool(daemonThreadFactory(settings, "transport_server_worker"))));
305            } else {
306                serverBootstrap = new ServerBootstrap(new NioServerSocketChannelFactory(
307                        Executors.newCachedThreadPool(daemonThreadFactory(settings, "transport_server_boss")),
308                        Executors.newCachedThreadPool(daemonThreadFactory(settings, "transport_server_worker")),
309                        workerCount));
310            }
311            ChannelPipelineFactory serverPipelineFactory = new ChannelPipelineFactory() {
312                @Override
313                public ChannelPipeline getPipeline() throws Exception {
314                    ChannelPipeline pipeline = Channels.pipeline();
315                    pipeline.addLast("openChannels", serverOpenChannels);
316                    SizeHeaderFrameDecoder sizeHeader = new SizeHeaderFrameDecoder();
317                    if (maxCumulationBufferCapacity != null) {
318                        if (maxCumulationBufferCapacity.bytes() > Integer.MAX_VALUE) {
319                            sizeHeader.setMaxCumulationBufferCapacity(Integer.MAX_VALUE);
320                        } else {
321                            sizeHeader.setMaxCumulationBufferCapacity((int) maxCumulationBufferCapacity.bytes());
322                        }
323                    }
324                    if (maxCompositeBufferComponents != -1) {
325                        sizeHeader.setMaxCumulationBufferComponents(maxCompositeBufferComponents);
326                    }
327                    pipeline.addLast("size", sizeHeader);
328                    pipeline.addLast("dispatcher", new MessageChannelHandler(NettyTransport.this, logger));
329                    return pipeline;
330                }
331            };
332            serverBootstrap.setPipelineFactory(serverPipelineFactory);
333            if (tcpNoDelay != null) {
334                serverBootstrap.setOption("child.tcpNoDelay", tcpNoDelay);
335            }
336            if (tcpKeepAlive != null) {
337                serverBootstrap.setOption("child.keepAlive", tcpKeepAlive);
338            }
339            if (tcpSendBufferSize != null && tcpSendBufferSize.bytes() > 0) {
340                serverBootstrap.setOption("child.sendBufferSize", tcpSendBufferSize.bytes());
341            }
342            if (tcpReceiveBufferSize != null && tcpReceiveBufferSize.bytes() > 0) {
343                serverBootstrap.setOption("child.receiveBufferSize", tcpReceiveBufferSize.bytes());
344            }
345            serverBootstrap.setOption("receiveBufferSizePredictorFactory", receiveBufferSizePredictorFactory);
346            serverBootstrap.setOption("child.receiveBufferSizePredictorFactory", receiveBufferSizePredictorFactory);
347            if (reuseAddress != null) {
348                serverBootstrap.setOption("reuseAddress", reuseAddress);
349                serverBootstrap.setOption("child.reuseAddress", reuseAddress);
350            }
351    
352            // Bind and start to accept incoming connections.
353            InetAddress hostAddressX;
354            try {
355                hostAddressX = networkService.resolveBindHostAddress(bindHost);
356            } catch (IOException e) {
357                throw new BindTransportException("Failed to resolve host [" + bindHost + "]", e);
358            }
359            final InetAddress hostAddress = hostAddressX;
360    
361            PortsRange portsRange = new PortsRange(port);
362            final AtomicReference<Exception> lastException = new AtomicReference();
363            boolean success = portsRange.iterate(new PortsRange.PortCallback() {
364                @Override
365                public boolean onPortNumber(int portNumber) {
366                    try {
367                        serverChannel = serverBootstrap.bind(new InetSocketAddress(hostAddress, portNumber));
368                    } catch (Exception e) {
369                        lastException.set(e);
370                        return false;
371                    }
372                    return true;
373                }
374            });
375            if (!success) {
376                throw new BindTransportException("Failed to bind to [" + port + "]", lastException.get());
377            }
378    
379            logger.debug("Bound to address [{}]", serverChannel.getLocalAddress());
380    
381            InetSocketAddress boundAddress = (InetSocketAddress) serverChannel.getLocalAddress();
382            InetSocketAddress publishAddress;
383            try {
384                publishAddress = new InetSocketAddress(networkService.resolvePublishHostAddress(publishHost), boundAddress.getPort());
385            } catch (Exception e) {
386                throw new BindTransportException("Failed to resolve publish address", e);
387            }
388            this.boundAddress = new BoundTransportAddress(new InetSocketTransportAddress(boundAddress), new InetSocketTransportAddress(publishAddress));
389        }

391        @Override
392        protected void doStop() throws ElasticsearchException {
393            final CountDownLatch latch = new CountDownLatch(1);
394            // make sure we run it on another thread than a possible IO handler thread
395            threadPool.generic().execute(new Runnable() {
396                @Override
397                public void run() {
398                    try {
399                        for (Iterator<NodeChannels> it = connectedNodes.values().iterator(); it.hasNext();) {
400                            NodeChannels nodeChannels = it.next();
401                            it.remove();
402                            nodeChannels.close();
403                        }
404    
405                        if (serverChannel != null) {
406                            try {
407                                serverChannel.close().awaitUninterruptibly();
408                            } finally {
409                                serverChannel = null;
410                            }
411                        }
412    
413                        if (serverOpenChannels != null) {
414                            serverOpenChannels.close();
415                            serverOpenChannels = null;
416                        }
417    
418                        if (serverBootstrap != null) {
419                            serverBootstrap.releaseExternalResources();
420                            serverBootstrap = null;
421                        }
422    
423                        for (Iterator<NodeChannels> it = connectedNodes.values().iterator(); it.hasNext();) {
424                            NodeChannels nodeChannels = it.next();
425                            it.remove();
426                            nodeChannels.close();
427                        }
428    
429                        if (clientBootstrap != null) {
430                            clientBootstrap.releaseExternalResources();
431                            clientBootstrap = null;
432                        }
433                    } finally {
434                        latch.countDown();
435                    }
436                }
437            });
438    
439            try {
440                latch.await(30, TimeUnit.SECONDS);
441            } catch (InterruptedException e) {
442                // ignore
443            }
444        }

450        @Override
451        public TransportAddress[] addressesFromString(String address) throws Exception {
452            int index = address.indexOf('[');
453            if (index != -1) {
454                String host = address.substring(0, index);
455                Set<String> ports = Strings.commaDelimitedListToSet(address.substring(index + 1, address.indexOf(']')));
456                List<TransportAddress> addresses = Lists.newArrayList();
457                for (String port : ports) {
458                    int[] iPorts = new PortsRange(port).ports();
459                    for (int iPort : iPorts) {
460                        addresses.add(new InetSocketTransportAddress(host, iPort));
461                    }
462                }
463                return addresses.toArray(new TransportAddress[addresses.size()]);
464            } else {
465                index = address.lastIndexOf(':');
466                if (index == -1) {
467                    List<TransportAddress> addresses = Lists.newArrayList();
468                    int[] iPorts = new PortsRange(this.port).ports();
469                    for (int iPort : iPorts) {
470                        addresses.add(new InetSocketTransportAddress(address, iPort));
471                    }
472                    return addresses.toArray(new TransportAddress[addresses.size()]);
473                } else {
474                    String host = address.substring(0, index);
475                    int port = Integer.parseInt(address.substring(index + 1));
476                    return new TransportAddress[]{new InetSocketTransportAddress(host, port)};
477                }
478            }
479        }

491        void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e) throws Exception {
492            if (!lifecycle.started()) {
493                // ignore
494            }
495            if (isCloseConnectionException(e.getCause())) {
496                // disconnect the node
497                Channel channel = ctx.getChannel();
498                for (Map.Entry<DiscoveryNode, NodeChannels> entry : connectedNodes.entrySet()) {
499                    if (entry.getValue().hasChannel(channel)) {
500                        disconnectFromNode(entry.getKey());
501                    }
502                }
503            } else if (isConnectException(e.getCause())) {
504                if (logger.isTraceEnabled()) {
505                    logger.trace("(Ignoring) Exception caught on netty layer [" + ctx.getChannel() + "]", e.getCause());
506                }
507            } else {
508                logger.warn("Exception caught on netty layer [" + ctx.getChannel() + "]", e.getCause());
509            }
510        }

522        @Override
523        public void sendRequest(final DiscoveryNode node, final long requestId, final String action, final TransportRequest request, TransportRequestOptions options) throws IOException, TransportException {
524            Channel targetChannel = nodeChannel(node, options);
525    
526            if (compress) {
527                options.withCompress(true);
528            }
529    
530            byte status = 0;
531            status = TransportStatus.setRequest(status);
532    
533            BytesStreamOutput bStream = new BytesStreamOutput();
534            bStream.skip(NettyHeader.HEADER_SIZE);
535            StreamOutput stream = bStream;
536            stream = new HandlesStreamOutput(stream);
537    
538            // we pick the smallest of the 2, to support both backward and forward compatibility
539            // note, this is the only place we need to do this, since from here on, we use the serialized version
540            // as the version to use also when the node receiving this request will send the response with
541            Version version = Version.smallest(this.version, node.version());
542    
543            stream.setVersion(version);
544            stream.writeString(action);
545    
546            ChannelBuffer buffer;
547            request.writeTo(stream);
548            stream.close();
549            buffer = bStream.ourBytes().toChannelBuffer();
550            NettyHeader.writeHeader(buffer, requestId, status, version);
551            targetChannel.write(buffer);
552        }

569        public void connectToNode(DiscoveryNode node, boolean light) {
570            if (!lifecycle.started()) {
571                throw new ElasticsearchIllegalStateException("can't add nodes to a stopped transport");
572            }
573            if (node == null) {
574                throw new ConnectTransportException(null, "can't connect to a null node");
575            }
576            globalLock.readLock().lock();
577            try {
578                if (!lifecycle.started()) {
579                    throw new ElasticsearchIllegalStateException("can't add nodes to a stopped transport");
580                }
581                NodeChannels nodeChannels = connectedNodes.get(node);
582                if (nodeChannels != null) {
583                    return;
584                }
585                connectionLock.acquire(node.id());
586                try {
587                    if (!lifecycle.started()) {
588                        throw new ElasticsearchIllegalStateException("can't add nodes to a stopped transport");
589                    }
590                    try {
591    
592    
593                        if (light) {
594                            nodeChannels = connectToChannelsLight(node);
595                        } else {
596                            nodeChannels = new NodeChannels(new Channel[connectionsPerNodeRecovery], new Channel[connectionsPerNodeBulk], new Channel[connectionsPerNodeReg], new Channel[connectionsPerNodeState], new Channel[connectionsPerNodePing]);
597                            try {
598                                connectToChannels(nodeChannels, node);
599                            } catch (Exception e) {
600                                nodeChannels.close();
601                                throw e;
602                            }
603                        }
604    
605                        NodeChannels existing = connectedNodes.putIfAbsent(node, nodeChannels);
606                        if (existing != null) {
607                            // we are already connected to a node, close this ones
608                            nodeChannels.close();
609                        } else {
610                            if (logger.isDebugEnabled()) {
611                                logger.debug("connected to node [{}]", node);
612                            }
613                            transportServiceAdapter.raiseNodeConnected(node);
614                        }
615    
616                    } catch (ConnectTransportException e) {
617                        throw e;
618                    } catch (Exception e) {
619                        throw new ConnectTransportException(node, "General node connection failure", e);
620                    }
621                } finally {
622                    connectionLock.release(node.id());
623                }
624            } finally {
625                globalLock.readLock().unlock();
626            }
627        }
628    
629        private NodeChannels connectToChannelsLight(DiscoveryNode node) {
630            InetSocketAddress address = ((InetSocketTransportAddress) node.address()).address();
631            ChannelFuture connect = clientBootstrap.connect(address);
632            connect.awaitUninterruptibly((long) (connectTimeout.millis() * 1.5));
633            if (!connect.isSuccess()) {
634                throw new ConnectTransportException(node, "connect_timeout[" + connectTimeout + "]", connect.getCause());
635            }
636            Channel[] channels = new Channel[1];
637            channels[0] = connect.getChannel();
638            channels[0].getCloseFuture().addListener(new ChannelCloseListener(node));
639            return new NodeChannels(channels, channels, channels, channels, channels);
640        }

642        private void connectToChannels(NodeChannels nodeChannels, DiscoveryNode node) {
643            ChannelFuture[] connectRecovery = new ChannelFuture[nodeChannels.recovery.length];
644            ChannelFuture[] connectBulk = new ChannelFuture[nodeChannels.bulk.length];
645            ChannelFuture[] connectReg = new ChannelFuture[nodeChannels.reg.length];
646            ChannelFuture[] connectState = new ChannelFuture[nodeChannels.state.length];
647            ChannelFuture[] connectPing = new ChannelFuture[nodeChannels.ping.length];
648            InetSocketAddress address = ((InetSocketTransportAddress) node.address()).address();
649            for (int i = 0; i < connectRecovery.length; i++) {
650                connectRecovery[i] = clientBootstrap.connect(address);
651            }
652            for (int i = 0; i < connectBulk.length; i++) {
653                connectBulk[i] = clientBootstrap.connect(address);
654            }
655            for (int i = 0; i < connectReg.length; i++) {
656                connectReg[i] = clientBootstrap.connect(address);
657            }
658            for (int i = 0; i < connectState.length; i++) {
659                connectState[i] = clientBootstrap.connect(address);
660            }
661            for (int i = 0; i < connectPing.length; i++) {
662                connectPing[i] = clientBootstrap.connect(address);
663            }
664    
665            try {
666                for (int i = 0; i < connectRecovery.length; i++) {
667                    connectRecovery[i].awaitUninterruptibly((long) (connectTimeout.millis() * 1.5));
668                    if (!connectRecovery[i].isSuccess()) {
669                        throw new ConnectTransportException(node, "connect_timeout[" + connectTimeout + "]", connectRecovery[i].getCause());
670                    }
671                    nodeChannels.recovery[i] = connectRecovery[i].getChannel();
672                    nodeChannels.recovery[i].getCloseFuture().addListener(new ChannelCloseListener(node));
673                }
674    
675                for (int i = 0; i < connectBulk.length; i++) {
676                    connectBulk[i].awaitUninterruptibly((long) (connectTimeout.millis() * 1.5));
677                    if (!connectBulk[i].isSuccess()) {
678                        throw new ConnectTransportException(node, "connect_timeout[" + connectTimeout + "]", connectBulk[i].getCause());
679                    }
680                    nodeChannels.bulk[i] = connectBulk[i].getChannel();
681                    nodeChannels.bulk[i].getCloseFuture().addListener(new ChannelCloseListener(node));
682                }
683    
684                for (int i = 0; i < connectReg.length; i++) {
685                    connectReg[i].awaitUninterruptibly((long) (connectTimeout.millis() * 1.5));
686                    if (!connectReg[i].isSuccess()) {
687                        throw new ConnectTransportException(node, "connect_timeout[" + connectTimeout + "]", connectReg[i].getCause());
688                    }
689                    nodeChannels.reg[i] = connectReg[i].getChannel();
690                    nodeChannels.reg[i].getCloseFuture().addListener(new ChannelCloseListener(node));
691                }
692    
693                for (int i = 0; i < connectState.length; i++) {
694                    connectState[i].awaitUninterruptibly((long) (connectTimeout.millis() * 1.5));
695                    if (!connectState[i].isSuccess()) {
696                        throw new ConnectTransportException(node, "connect_timeout[" + connectTimeout + "]", connectState[i].getCause());
697                    }
698                    nodeChannels.state[i] = connectState[i].getChannel();
699                    nodeChannels.state[i].getCloseFuture().addListener(new ChannelCloseListener(node));
700                }
701    
702                for (int i = 0; i < connectPing.length; i++) {
703                    connectPing[i].awaitUninterruptibly((long) (connectTimeout.millis() * 1.5));
704                    if (!connectPing[i].isSuccess()) {
705                        throw new ConnectTransportException(node, "connect_timeout[" + connectTimeout + "]", connectPing[i].getCause());
706                    }
707                    nodeChannels.ping[i] = connectPing[i].getChannel();
708                    nodeChannels.ping[i].getCloseFuture().addListener(new ChannelCloseListener(node));
709                }
710    
711                if (nodeChannels.recovery.length == 0) {
712                    if (nodeChannels.bulk.length > 0) {
713                        nodeChannels.recovery = nodeChannels.bulk;
714                    } else {
715                        nodeChannels.recovery = nodeChannels.reg;
716                    }
717                }
718                if (nodeChannels.bulk.length == 0) {
719                    nodeChannels.bulk = nodeChannels.reg;
720                }
721            } catch (RuntimeException e) {
722                // clean the futures
723                for (ChannelFuture future : ImmutableList.<ChannelFuture>builder().add(connectRecovery).add(connectBulk).add(connectReg).add(connectState).add(connectPing).build()) {
724                    future.cancel();
725                    if (future.getChannel() != null && future.getChannel().isOpen()) {
726                        try {
727                            future.getChannel().close();
728                        } catch (Exception e1) {
729                            // ignore
730                        }
731                    }
732                }
733                throw e;
734            }
735        }

737        @Override
738        public void disconnectFromNode(DiscoveryNode node) {
739            NodeChannels nodeChannels = connectedNodes.remove(node);
740            if (nodeChannels != null) {
741                connectionLock.acquire(node.id());
742                try {
743                    try {
744                        nodeChannels.close();
745                    } finally {
746                        logger.debug("disconnected from [{}]", node);
747                        transportServiceAdapter.raiseNodeDisconnected(node);
748                    }
749                } finally {
750                    connectionLock.release(node.id());
751                }
752            }
753        }

755        /**
756         * Disconnects from a node if a channel is found as part of that nodes channels.
757         */
758        private void disconnectFromNodeChannel(Channel channel, Throwable failure) {
759            for (DiscoveryNode node : connectedNodes.keySet()) {
760                NodeChannels nodeChannels = connectedNodes.get(node);
761                if (nodeChannels != null && nodeChannels.hasChannel(channel)) {
762                    connectionLock.acquire(node.id());
763                    if (!nodeChannels.hasChannel(channel)) { //might have been removed in the meanwhile, safety check
764                        assert !connectedNodes.containsKey(node);
765                    } else {
766                        try {
767                            connectedNodes.remove(node);
768                            try {
769                                nodeChannels.close();
770                            } finally {
771                                logger.debug("disconnected from [{}] on channel failure", failure, node);
772                                transportServiceAdapter.raiseNodeDisconnected(node);
773                            }
774                        } finally {
775                            connectionLock.release(node.id());
776                        }
777                    }
778                }
779            }
780        }

782        private Channel nodeChannel(DiscoveryNode node, TransportRequestOptions options) throws ConnectTransportException {
783            NodeChannels nodeChannels = connectedNodes.get(node);
784            if (nodeChannels == null) {
785                throw new NodeNotConnectedException(node, "Node not connected");
786            }
787            return nodeChannels.channel(options.type());
788        }

790        private class ChannelCloseListener implements ChannelFutureListener {
791    
792            private final DiscoveryNode node;
793    
794            private ChannelCloseListener(DiscoveryNode node) {
795                this.node = node;
796            }
797    
798            @Override
799            public void operationComplete(ChannelFuture future) throws Exception {
800                disconnectFromNode(node);
801            }

804        public static class NodeChannels {
805    
806            private Channel[] recovery;
807            private final AtomicInteger recoveryCounter = new AtomicInteger();
808            private Channel[] bulk;
809            private final AtomicInteger bulkCounter = new AtomicInteger();
810            private Channel[] reg;
811            private final AtomicInteger regCounter = new AtomicInteger();
812            private Channel[] state;
813            private final AtomicInteger stateCounter = new AtomicInteger();
814            private Channel[] ping;
815            private final AtomicInteger pingCounter = new AtomicInteger();
816    
817            public NodeChannels(Channel[] recovery, Channel[] bulk, Channel[] reg, Channel[] state, Channel[] ping) {
818                this.recovery = recovery;
819                this.bulk = bulk;
820                this.reg = reg;
821                this.state = state;
822                this.ping = ping;
823            }
824    
825            public boolean hasChannel(Channel channel) {
826                return hasChannel(channel, recovery) || hasChannel(channel, bulk) || hasChannel(channel, reg) || hasChannel(channel, state) || hasChannel(channel, ping);
827            }
828    
829            private boolean hasChannel(Channel channel, Channel[] channels) {
830                for (Channel channel1 : channels) {
831                    if (channel.equals(channel1)) {
832                        return true;
833                    }
834                }
835                return false;
836            }
837    
838            public Channel channel(TransportRequestOptions.Type type) {
839                if (type == TransportRequestOptions.Type.REG) {
840                    return reg[Math.abs(regCounter.incrementAndGet()) % reg.length];
841                } else if (type == TransportRequestOptions.Type.STATE) {
842                    return state[Math.abs(stateCounter.incrementAndGet()) % state.length];
843                } else if (type == TransportRequestOptions.Type.PING) {
844                    return ping[Math.abs(pingCounter.incrementAndGet()) % ping.length];
845                } else if (type == TransportRequestOptions.Type.BULK) {
846                    return bulk[Math.abs(bulkCounter.incrementAndGet()) % bulk.length];
847                } else if (type == TransportRequestOptions.Type.RECOVERY) {
848                    return recovery[Math.abs(recoveryCounter.incrementAndGet()) % recovery.length];
849                } else {
850                    throw new ElasticsearchIllegalArgumentException("no type channel for [" + type + "]");
851                }
852            }
853    
854            public synchronized void close() {
855                List<ChannelFuture> futures = new ArrayList<ChannelFuture>();
856                closeChannelsAndWait(recovery, futures);
857                closeChannelsAndWait(bulk, futures);
858                closeChannelsAndWait(reg, futures);
859                closeChannelsAndWait(state, futures);
860                closeChannelsAndWait(ping, futures);
861                for (ChannelFuture future : futures) {
862                    future.awaitUninterruptibly();
863                }
864            }
865    
866            private void closeChannelsAndWait(Channel[] channels, List<ChannelFuture> futures) {
867                for (Channel channel : channels) {
868                    try {
869                        if (channel != null && channel.isOpen()) {
870                            futures.add(channel.close());
871                        }
872                    } catch (Exception e) {
873                        //ignore
874                    }
875                }
876            }
877        }
