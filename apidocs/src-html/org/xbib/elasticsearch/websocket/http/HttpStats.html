001    
002    package org.xbib.elasticsearch.websocket.http;
003    
004    import java.io.IOException;
005    
006    import org.elasticsearch.common.io.stream.StreamInput;
007    import org.elasticsearch.common.io.stream.StreamOutput;
008    import org.elasticsearch.common.io.stream.Streamable;
009    import org.elasticsearch.common.xcontent.ToXContent;
010    import org.elasticsearch.common.xcontent.XContentBuilder;
011    import org.elasticsearch.common.xcontent.XContentBuilderString;
012    
013    public class HttpStats implements Streamable, ToXContent {
014    
015        private long serverOpen;
016        private long totalOpen;
017    
018        HttpStats() {
019        }
020    
021        public HttpStats(long serverOpen, long totalOpen) {
022            this.serverOpen = serverOpen;
023            this.totalOpen = totalOpen;
024        }
025    
026        public long serverOpen() {
027            return this.serverOpen;
028        }
029    
030        public long getServerOpen() {
031            return serverOpen();
032        }
033    
034        public long totalOpen() {
035            return this.totalOpen;
036        }
037    
038        public long getTotalOpen() {
039            return this.totalOpen;
040        }
041    
042        public static HttpStats readHttpStats(StreamInput in) throws IOException {
043            HttpStats stats = new HttpStats();
044            stats.readFrom(in);
045            return stats;
046        }
047    
048        @Override
049        public void readFrom(StreamInput in) throws IOException {
050            serverOpen = in.readVLong();
051            totalOpen = in.readVLong();
052        }
053    
054        @Override
055        public void writeTo(StreamOutput out) throws IOException {
056            out.writeVLong(serverOpen);
057            out.writeVLong(totalOpen);
058        }
059    
060        static final class Fields {
061            static final XContentBuilderString WEBSOCKET = new XContentBuilderString("websocket");
062            static final XContentBuilderString CURRENT_OPEN = new XContentBuilderString("current_open");
063            static final XContentBuilderString TOTAL_OPENED = new XContentBuilderString("total_opened");
064        }
065    
066        @Override
067        public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
068            builder.startObject(Fields.WEBSOCKET);
069            builder.field(Fields.CURRENT_OPEN, serverOpen);
070            builder.field(Fields.TOTAL_OPENED, totalOpen);
071            builder.endObject();
072            return builder;
073        }
074    }
