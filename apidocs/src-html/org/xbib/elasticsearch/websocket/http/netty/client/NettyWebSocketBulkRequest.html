002    package org.xbib.elasticsearch.websocket.http.netty.client;
003    
004    import java.io.IOException;
005    
006    import org.elasticsearch.common.xcontent.XContentBuilder;
007    
008    import org.xbib.elasticsearch.websocket.client.WebSocketClient;
009    import org.xbib.elasticsearch.websocket.client.WebSocketClientBulkRequest;
010    import org.xbib.elasticsearch.websocket.http.netty.NettyInteractiveRequest;
011    import org.xbib.elasticsearch.websocket.http.netty.NettyInteractiveResponse;
012    
013    /**
014     * Netty bulk request convenience class.
015     */
016    public class NettyWebSocketBulkRequest 
017        extends NettyInteractiveRequest
018        implements WebSocketClientBulkRequest {
019        
020        public NettyWebSocketBulkRequest(String type) {
021            super.type(type);
022        }
023        
024        @Override
025        public NettyWebSocketBulkRequest data(XContentBuilder builder) {
026            super.data(builder);
027            return this;
028        }
029    
030        @Override
031        public void send(WebSocketClient client) throws IOException {
032            client.send(new NettyInteractiveResponse(super.type, super.builder).response());
033        }
034    
035    }
