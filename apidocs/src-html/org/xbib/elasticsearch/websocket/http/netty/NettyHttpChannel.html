004    import java.io.IOException;
005    import java.util.Set;
006

008    import org.elasticsearch.rest.RestStatus;
009    import org.elasticsearch.rest.XContentRestResponse;
010    import org.elasticsearch.rest.support.RestUtils;

028    /**
029     *
030     */
031    public class NettyHttpChannel implements HttpChannel {
032    
033        private final NettyWebSocketServerTransport transport;
034    
035        private final Channel channel;
036    
037        private final org.jboss.netty.handler.codec.http.HttpRequest request;
038    
039        public NettyHttpChannel(NettyWebSocketServerTransport transport, Channel channel,
040                                org.jboss.netty.handler.codec.http.HttpRequest request) {
041            this.transport = transport;
042            this.channel = channel;
043            this.request = request;
044        }
045    
046        @Override
047        public void sendResponse(RestResponse response) {
048    
049            // Decide whether to close the connection or not.
050            boolean http10 = request.getProtocolVersion().equals(HttpVersion.HTTP_1_0);
051            boolean close =
052                    HttpHeaders.Values.CLOSE.equalsIgnoreCase(request.headers().get(HttpHeaders.Names.CONNECTION)) ||
053                            (http10 && !HttpHeaders.Values.KEEP_ALIVE.equalsIgnoreCase(request.headers().get(HttpHeaders.Names.CONNECTION)));
054    
055            // Build the response object.
056            HttpResponseStatus status = getStatus(response.status());
057            org.jboss.netty.handler.codec.http.HttpResponse resp;
058            if (http10) {
059                resp = new DefaultHttpResponse(HttpVersion.HTTP_1_0, status);
060                if (!close) {
061                    resp.headers().add(HttpHeaders.Names.CONNECTION, "Keep-Alive");
062                }
063            } else {
064                resp = new DefaultHttpResponse(HttpVersion.HTTP_1_1, status);
065            }
066            if (RestUtils.isBrowser(request.headers().get(HttpHeaders.Names.USER_AGENT))) {
067                // add support for cross origin
068                resp.headers().add("Access-Control-Allow-Origin", "*");
069                if (request.getMethod() == HttpMethod.OPTIONS) {
070                    // also add more access control parameters
071                    resp.headers().add("Access-Control-Max-Age", 1728000);
072                    resp.headers().add("Access-Control-Allow-Methods", "PUT, DELETE");
073                    resp.headers().add("Access-Control-Allow-Headers", "X-Requested-With");
074                }
075            }
076    
077            String opaque = request.headers().get("X-Opaque-Id");
078            if (opaque != null) {
079                resp.headers().add("X-Opaque-Id", opaque);
080            }
081    
082            // Convert the response content to a ChannelBuffer.
083            ChannelFutureListener releaseContentListener = null;
084            ChannelBuffer buf;
085            try {
086                if (response instanceof XContentRestResponse) {
087                    if (response.contentThreadSafe()) {
088                        buf = ChannelBuffers.wrappedBuffer(response.content(), 0, response.contentLength());
089                    } else {
090                        buf = ChannelBuffers.copiedBuffer(response.content(), 0, response.contentLength());
091                    }
092                } else {
093                    if (response.contentThreadSafe()) {
094                        buf = ChannelBuffers.wrappedBuffer(response.content(), 0, response.contentLength());
095                    } else {
096                        buf = ChannelBuffers.copiedBuffer(response.content(), 0, response.contentLength());
097                    }
098                }
099            } catch (IOException e) {
100                throw new HttpException(e.getMessage(), e);
101            }
102            if (response.prefixContent() != null || response.suffixContent() != null) {
103                ChannelBuffer prefixBuf = ChannelBuffers.EMPTY_BUFFER;
104                if (response.prefixContent() != null) {
105                    prefixBuf = ChannelBuffers.copiedBuffer(response.prefixContent(), 0, response.prefixContentLength());
106                }
107                ChannelBuffer suffixBuf = ChannelBuffers.EMPTY_BUFFER;
108                if (response.suffixContent() != null) {
109                    suffixBuf = ChannelBuffers.copiedBuffer(response.suffixContent(), 0, response.suffixContentLength());
110                }
111                buf = ChannelBuffers.wrappedBuffer(prefixBuf, buf, suffixBuf);
112            }
113    
114            resp.headers().set(HttpHeaders.Names.CONTENT_TYPE, response.contentType());
115            resp.headers().set(HttpHeaders.Names.CONTENT_LENGTH, String.valueOf(buf.readableBytes()));
116    
117            if (transport.resetCookies) {
118                String cookieString = request.headers().get(HttpHeaders.Names.COOKIE);
119                if (cookieString != null) {
120                    CookieDecoder cookieDecoder = new CookieDecoder();
121                    Set<Cookie> cookies = cookieDecoder.decode(cookieString);
122                    if (!cookies.isEmpty()) {
123                        CookieEncoder cookieEncoder = new CookieEncoder(true);
124                        for (Cookie cookie : cookies) {
125                            cookieEncoder.addCookie(cookie);
126                        }
127                        resp.headers().add(HttpHeaders.Names.SET_COOKIE, cookieEncoder.encode());
128                    }
129                }
130            }
131    
132            // Write the response.
133            ChannelFuture future = channel.write(resp);
134            if (releaseContentListener != null) {
135                future.addListener(releaseContentListener);
136            }
137    
138            // Close the connection after the write operation is done if necessary.
139            if (close) {
140                future.addListener(ChannelFutureListener.CLOSE);
141            }
142        }
143    
144        private HttpResponseStatus getStatus(RestStatus status) {
145            switch (status) {
146                case CONTINUE:
147                    return HttpResponseStatus.CONTINUE;
148                case SWITCHING_PROTOCOLS:
149                    return HttpResponseStatus.SWITCHING_PROTOCOLS;
150                case OK:
151                    return HttpResponseStatus.OK;
152                case CREATED:
153                    return HttpResponseStatus.CREATED;
154                case ACCEPTED:
155                    return HttpResponseStatus.ACCEPTED;
156                case NON_AUTHORITATIVE_INFORMATION:
157                    return HttpResponseStatus.NON_AUTHORITATIVE_INFORMATION;
158                case NO_CONTENT:
159                    return HttpResponseStatus.NO_CONTENT;
160                case RESET_CONTENT:
161                    return HttpResponseStatus.RESET_CONTENT;
162                case PARTIAL_CONTENT:
163                    return HttpResponseStatus.PARTIAL_CONTENT;
164                case MULTI_STATUS:
165                    // no status for this??
166                    return HttpResponseStatus.INTERNAL_SERVER_ERROR;
167                case MULTIPLE_CHOICES:
168                    return HttpResponseStatus.MULTIPLE_CHOICES;
169                case MOVED_PERMANENTLY:
170                    return HttpResponseStatus.MOVED_PERMANENTLY;
171                case FOUND:
172                    return HttpResponseStatus.FOUND;
173                case SEE_OTHER:
174                    return HttpResponseStatus.SEE_OTHER;
175                case NOT_MODIFIED:
176                    return HttpResponseStatus.NOT_MODIFIED;
177                case USE_PROXY:
178                    return HttpResponseStatus.USE_PROXY;
179                case TEMPORARY_REDIRECT:
180                    return HttpResponseStatus.TEMPORARY_REDIRECT;
181                case BAD_REQUEST:
182                    return HttpResponseStatus.BAD_REQUEST;
183                case UNAUTHORIZED:
184                    return HttpResponseStatus.UNAUTHORIZED;
185                case PAYMENT_REQUIRED:
186                    return HttpResponseStatus.PAYMENT_REQUIRED;
187                case FORBIDDEN:
188                    return HttpResponseStatus.FORBIDDEN;
189                case NOT_FOUND:
190                    return HttpResponseStatus.NOT_FOUND;
191                case METHOD_NOT_ALLOWED:
192                    return HttpResponseStatus.METHOD_NOT_ALLOWED;
193                case NOT_ACCEPTABLE:
194                    return HttpResponseStatus.NOT_ACCEPTABLE;
195                case PROXY_AUTHENTICATION:
196                    return HttpResponseStatus.PROXY_AUTHENTICATION_REQUIRED;
197                case REQUEST_TIMEOUT:
198                    return HttpResponseStatus.REQUEST_TIMEOUT;
199                case CONFLICT:
200                    return HttpResponseStatus.CONFLICT;
201                case GONE:
202                    return HttpResponseStatus.GONE;
203                case LENGTH_REQUIRED:
204                    return HttpResponseStatus.LENGTH_REQUIRED;
205                case PRECONDITION_FAILED:
206                    return HttpResponseStatus.PRECONDITION_FAILED;
207                case REQUEST_ENTITY_TOO_LARGE:
208                    return HttpResponseStatus.REQUEST_ENTITY_TOO_LARGE;
209                case REQUEST_URI_TOO_LONG:
210                    return HttpResponseStatus.REQUEST_URI_TOO_LONG;
211                case UNSUPPORTED_MEDIA_TYPE:
212                    return HttpResponseStatus.UNSUPPORTED_MEDIA_TYPE;
213                case REQUESTED_RANGE_NOT_SATISFIED:
214                    return HttpResponseStatus.REQUESTED_RANGE_NOT_SATISFIABLE;
215                case EXPECTATION_FAILED:
216                    return HttpResponseStatus.EXPECTATION_FAILED;
217                case UNPROCESSABLE_ENTITY:
218                    return HttpResponseStatus.BAD_REQUEST;
219                case LOCKED:
220                    return HttpResponseStatus.BAD_REQUEST;
221                case FAILED_DEPENDENCY:
222                    return HttpResponseStatus.BAD_REQUEST;
223                case INTERNAL_SERVER_ERROR:
224                    return HttpResponseStatus.INTERNAL_SERVER_ERROR;
225                case NOT_IMPLEMENTED:
226                    return HttpResponseStatus.NOT_IMPLEMENTED;
227                case BAD_GATEWAY:
228                    return HttpResponseStatus.BAD_GATEWAY;
229                case SERVICE_UNAVAILABLE:
230                    return HttpResponseStatus.SERVICE_UNAVAILABLE;
231                case GATEWAY_TIMEOUT:
232                    return HttpResponseStatus.GATEWAY_TIMEOUT;
233                case HTTP_VERSION_NOT_SUPPORTED:
234                    return HttpResponseStatus.HTTP_VERSION_NOT_SUPPORTED;
235                default:
236                    return HttpResponseStatus.INTERNAL_SERVER_ERROR;
237            }
238        }
