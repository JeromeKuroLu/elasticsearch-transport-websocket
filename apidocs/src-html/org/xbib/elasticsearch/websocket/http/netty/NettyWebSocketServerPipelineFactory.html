002    package org.xbib.elasticsearch.websocket.http.netty;
003    
004    import org.jboss.netty.channel.ChannelPipeline;
005    
006    /**
007     * Netty implementation for a WebSocket server pipeline factory.
008     * It is based on the HTTP server pipeline factory.
009     */
010    public class NettyWebSocketServerPipelineFactory extends NettyHttpServerPipelineFactory {
011    
012        private final NettyWebSocketRequestHandler handler;
013        
014        public NettyWebSocketServerPipelineFactory(NettyWebSocketServerTransport transport) {
015            super(transport);
016            this.handler = new NettyWebSocketRequestHandler(transport);
017        }
018    
019        @Override
020        public ChannelPipeline getPipeline() throws Exception {
021            ChannelPipeline pipeline = super.getPipeline();
022            pipeline.replace("handler", "handler", handler);
023            return pipeline;
024        }
025    }
