002    package org.xbib.elasticsearch.websocket.http.netty;
003    
004    import org.jboss.netty.channel.ChannelHandlerContext;
005    import org.jboss.netty.channel.ExceptionEvent;
006    import org.jboss.netty.channel.MessageEvent;
007    import org.jboss.netty.handler.codec.http.HttpRequest;
008    import org.jboss.netty.handler.codec.http.websocketx.WebSocketFrame;
009    
010    /**
011     * Netty implementation for a WebSocket request handler.
012     * It is based on the HTTP request handler.
013     */
014    public class NettyWebSocketRequestHandler extends NettyHttpRequestHandler {
015    
016        public NettyWebSocketRequestHandler(NettyWebSocketServerTransport serverTransport) {
017            super(serverTransport);
018        }
019    
020        @Override
021        public void messageReceived(ChannelHandlerContext ctx, MessageEvent e) throws Exception {
022            Object msg = e.getMessage();
023            if (msg instanceof HttpRequest) {
024                super.messageReceived(ctx, e);
025            } else if (msg instanceof WebSocketFrame) {
026                serverTransport.frame(handshaker, (WebSocketFrame) msg, ctx);
027            }
028        }
029    
030        @Override
031        public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e) throws Exception {
032            serverTransport.exceptionCaught(ctx, e);
033            e.getChannel().close();
034        }
035    
036    }
