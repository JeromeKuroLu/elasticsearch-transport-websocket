002    package org.xbib.elasticsearch.websocket.http.netty;
003    
004    import org.jboss.netty.channel.ChannelPipeline;
005    import org.jboss.netty.channel.ChannelPipelineFactory;
006    import org.jboss.netty.channel.Channels;
007    import org.jboss.netty.handler.codec.http.HttpChunkAggregator;
008    import org.jboss.netty.handler.codec.http.HttpContentCompressor;
009    import org.jboss.netty.handler.codec.http.HttpContentDecompressor;
010    import org.jboss.netty.handler.codec.http.HttpRequestDecoder;
011    import org.jboss.netty.handler.codec.http.HttpResponseEncoder;
012    
013    public class NettyHttpServerPipelineFactory implements ChannelPipelineFactory {
014    
015        protected final NettyWebSocketServerTransport transport;
016    
017        private final NettyHttpRequestHandler requestHandler;
018    
019        NettyHttpServerPipelineFactory(NettyWebSocketServerTransport transport) {
020            this.transport = transport;
021            this.requestHandler = new NettyHttpRequestHandler(transport);
022        }
023    
024        @Override
025        public ChannelPipeline getPipeline() throws Exception {
026            ChannelPipeline pipeline = Channels.pipeline();
027            pipeline.addLast("openChannels", transport.serverOpenChannels);
028            HttpRequestDecoder requestDecoder = new HttpRequestDecoder(
029                    (int) transport.maxInitialLineLength.bytes(),
030                    (int) transport.maxHeaderSize.bytes(),
031                    (int) transport.maxChunkSize.bytes());
032            if (transport.maxCumulationBufferCapacity != null) {
033                if (transport.maxCumulationBufferCapacity.bytes() > Integer.MAX_VALUE) {
034                    requestDecoder.setMaxCumulationBufferCapacity(Integer.MAX_VALUE);
035                } else {
036                    requestDecoder.setMaxCumulationBufferCapacity((int) transport.maxCumulationBufferCapacity.bytes());
037                }
038            }
039            if (transport.maxCompositeBufferComponents != -1) {
040                requestDecoder.setMaxCumulationBufferComponents(transport.maxCompositeBufferComponents);
041            }
042            pipeline.addLast("decoder", requestDecoder);
043            if (transport.compression) {
044                pipeline.addLast("decoder_compress", new HttpContentDecompressor());
045            }
046            HttpChunkAggregator httpChunkAggregator = new HttpChunkAggregator((int) transport.maxContentLength.bytes());
047            if (transport.maxCompositeBufferComponents != -1) {
048                httpChunkAggregator.setMaxCumulationBufferComponents(transport.maxCompositeBufferComponents);
049            }
050            pipeline.addLast("aggregator", httpChunkAggregator);
051            pipeline.addLast("encoder", new HttpResponseEncoder());
052            if (transport.compression) {
053                pipeline.addLast("encoder_compress", new HttpContentCompressor(transport.compressionLevel));
054            }
055            pipeline.addLast("handler", requestHandler);
056            return pipeline;
057        }
058    }
