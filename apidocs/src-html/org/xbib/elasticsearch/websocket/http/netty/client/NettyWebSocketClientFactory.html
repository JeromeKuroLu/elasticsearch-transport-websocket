002    package org.xbib.elasticsearch.websocket.http.netty.client;
003    
004    import java.net.URI;
005    import java.util.concurrent.Executors;
006    
007    import org.jboss.netty.bootstrap.ClientBootstrap;
008    import org.jboss.netty.channel.ChannelPipeline;
009    import org.jboss.netty.channel.ChannelPipelineFactory;
010    import org.jboss.netty.channel.Channels;
011    import org.jboss.netty.channel.socket.nio.NioClientSocketChannelFactory;
012    import org.jboss.netty.handler.codec.http.HttpRequestEncoder;
013    import org.jboss.netty.handler.codec.http.HttpResponseDecoder;
014    
015    import org.xbib.elasticsearch.websocket.client.WebSocketActionListener;
016    import org.xbib.elasticsearch.websocket.client.WebSocketClient;
017    import org.xbib.elasticsearch.websocket.client.WebSocketClientBulkRequest;
018    import org.xbib.elasticsearch.websocket.client.WebSocketClientFactory;
019    import org.xbib.elasticsearch.websocket.client.WebSocketClientRequest;
020    import org.xbib.elasticsearch.websocket.http.netty.NettyInteractiveRequest;
021    
022    /**
023     * A factory for creating Websocket clients. The entry point for creating and
024     * connecting a client. Can and should be used to create multiple instances.
025     *
026     * Extended for Websocket client request methods.
027     */
028    public class NettyWebSocketClientFactory implements WebSocketClientFactory {
029    
030        private NioClientSocketChannelFactory socketChannelFactory = new NioClientSocketChannelFactory(
031                Executors.newCachedThreadPool(),
032                Executors.newCachedThreadPool());
033    
034        /**
035         * Create a new WebSocket client
036         *
037         * @param url URL to connect to.
038         * @param listener Callback interface to receive events
039         * @return A WebSocket client. Call {@link NettyWebSocketClient#connect()} to
040         * connect.
041         */
042        @Override
043        public WebSocketClient newClient(final URI url, final WebSocketActionListener listener) {
044            ClientBootstrap bootstrap = new ClientBootstrap(socketChannelFactory);
045    
046            String protocol = url.getScheme();
047            if (!protocol.equals("ws") && !protocol.equals("wss")) {
048                throw new IllegalArgumentException("unsupported protocol: " + protocol);
049            }
050    
051            final NettyWebSocketClientHandler clientHandler = new NettyWebSocketClientHandler(bootstrap, url, listener);
052    
053            bootstrap.setPipelineFactory(new ChannelPipelineFactory() {
054                @Override
055                public ChannelPipeline getPipeline() throws Exception {
056                    ChannelPipeline pipeline = Channels.pipeline();
057                    pipeline.addLast("decoder", new HttpResponseDecoder());
058                    pipeline.addLast("encoder", new HttpRequestEncoder());
059                    pipeline.addLast("ws-handler", clientHandler);
060                    return pipeline;
061                }
062            });
063    
064            return clientHandler;
065        }
066    
067        @Override
068        public void shutdown() {
069            socketChannelFactory.releaseExternalResources();
070        }
071    
072        @Override
073        public WebSocketClientRequest newRequest() {
074            return new NettyInteractiveRequest();
075        }
076    
077        @Override
078        public WebSocketClientBulkRequest indexRequest() {
079            return new NettyWebSocketBulkRequest("index");
080        }
081    
082        @Override
083        public WebSocketClientBulkRequest deleteRequest() {
084            return new NettyWebSocketBulkRequest("delete");
085        }
086    
087        @Override
088        public WebSocketClientRequest flushRequest() {
089            return new NettyInteractiveRequest().type("flush");
090        }
091    }
