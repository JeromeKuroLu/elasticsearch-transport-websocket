002    package org.xbib.elasticsearch.websocket.http.netty.client;
003    
004    import java.net.InetSocketAddress;
005    import java.net.URI;
006    
007    import org.jboss.netty.bootstrap.ClientBootstrap;
008    import org.jboss.netty.channel.Channel;
009    import org.jboss.netty.channel.ChannelFuture;
010    import org.jboss.netty.channel.ChannelHandlerContext;
011    import org.jboss.netty.channel.ChannelStateEvent;
012    import org.jboss.netty.channel.ExceptionEvent;
013    import org.jboss.netty.channel.MessageEvent;
014    import org.jboss.netty.channel.SimpleChannelUpstreamHandler;
015    import org.jboss.netty.handler.codec.http.DefaultHttpRequest;
016    import org.jboss.netty.handler.codec.http.HttpHeaders.Names;
017    import org.jboss.netty.handler.codec.http.HttpHeaders.Values;
018    import org.jboss.netty.handler.codec.http.HttpMethod;
019    import org.jboss.netty.handler.codec.http.HttpRequest;
020    import org.jboss.netty.handler.codec.http.HttpResponse;
021    import org.jboss.netty.handler.codec.http.HttpResponseStatus;
022    import org.jboss.netty.handler.codec.http.HttpVersion;
023    import org.jboss.netty.handler.codec.http.websocketx.WebSocket00FrameDecoder;
024    import org.jboss.netty.handler.codec.http.websocketx.WebSocket00FrameEncoder;
025    import org.jboss.netty.handler.codec.http.websocketx.WebSocketFrame;
026    import org.jboss.netty.util.CharsetUtil;
027    
028    import org.xbib.elasticsearch.websocket.client.WebSocketActionListener;
029    import org.xbib.elasticsearch.websocket.client.WebSocketClient;
030    
031    /**
032     * Handles socket communication for a connected WebSocket client. Not intended
033     * for end-users. Please use {@link NettyWebSocketClient} for controlling your client.
034     */
035    public class NettyWebSocketClientHandler extends SimpleChannelUpstreamHandler
036            implements WebSocketClient {
037    
038        private final ClientBootstrap bootstrap;
039        private final URI url;
040        private final WebSocketActionListener listener;
041        private boolean handshakeCompleted = false;
042        private Channel channel;
043    
044        public NettyWebSocketClientHandler(ClientBootstrap bootstrap, URI url, WebSocketActionListener listener) {
045            this.bootstrap = bootstrap;
046            this.url = url;
047            this.listener = listener;
048        }
049    
050        @Override
051        public Channel channel() {
052            return channel;
053        }
054    
055        @Override
056        public void channelConnected(ChannelHandlerContext ctx, ChannelStateEvent event) throws Exception {
057            String path = url.getPath();
058            if (url.getQuery() != null && url.getQuery().length() > 0) {
059                path = url.getPath() + "?" + url.getQuery();
060            }
061            HttpRequest request = new DefaultHttpRequest(HttpVersion.HTTP_1_1, HttpMethod.GET, path);
062            request.headers().add(Names.UPGRADE, Values.WEBSOCKET);
063            request.headers().add(Names.CONNECTION, Values.UPGRADE);
064            request.headers().add(Names.HOST, url.getHost());
065            request.headers().add(Names.ORIGIN, "http://" + url.getHost());
066            event.getChannel().write(request);
067            ctx.getPipeline().replace("encoder", "ws-encoder", new WebSocket00FrameEncoder());
068            this.channel = event.getChannel();
069        }
070    
071        @Override
072        public void channelClosed(ChannelHandlerContext ctx, ChannelStateEvent event) throws Exception {
073            listener.onDisconnect(this);
074            handshakeCompleted = false;
075            channel = null;
076        }
077    
078        @Override
079        public void messageReceived(ChannelHandlerContext ctx, MessageEvent event) throws Exception {
080            if (!handshakeCompleted) {
081                HttpResponse response = (HttpResponse) event.getMessage();
082                final HttpResponseStatus status = new HttpResponseStatus(101, "Web Socket Protocol Handshake");
083                final boolean validStatus = response.getStatus().equals(status);
084                final boolean validUpgrade = response.getHeader(Names.UPGRADE).equals(Values.WEBSOCKET);
085                final boolean validConnection = response.getHeader(Names.CONNECTION).equals(Values.UPGRADE);
086                if (!validStatus || !validUpgrade || !validConnection) {
087                    throw new NettyWebSocketException("Invalid handshake response");
088                }
089                handshakeCompleted = true;
090                ctx.getPipeline().replace("decoder", "ws-decoder", new WebSocket00FrameDecoder());
091                listener.onConnect(this);
092                return;
093            }
094            if (event.getMessage() instanceof HttpResponse) {
095                HttpResponse response = (HttpResponse) event.getMessage();
096                throw new NettyWebSocketException("Unexpected HttpResponse (status=" + response.getStatus() + ", content=" + response.getContent().toString(CharsetUtil.UTF_8) + ")");
097            }
098            WebSocketFrame frame = (WebSocketFrame) event.getMessage();
099            listener.onMessage(this, frame);
100        }
101    
102        @Override
103        public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e) throws Exception {
104            final Throwable t = e.getCause();
105            listener.onError(t);
106            e.getChannel().close();
107        }
108    
109        @Override
110        public ChannelFuture connect() {
111            return bootstrap.connect(new InetSocketAddress(url.getHost(), url.getPort()));
112        }
113    
114        @Override
115        public ChannelFuture disconnect() {
116            if (channel == null) {
117                return null;
118            }
119            return channel.close();
120        }
121    
122        @Override
123        public ChannelFuture send(WebSocketFrame frame) {
124            if (channel == null) {
125                return null;
126            }
127            return channel.write(frame);
128        }
129    
130    }
