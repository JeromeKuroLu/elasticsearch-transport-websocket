002    package org.xbib.elasticsearch.websocket.http;
003    
004    import org.jboss.netty.channel.Channel;
005    import org.jboss.netty.channel.ChannelHandlerContext;
006    import org.jboss.netty.handler.codec.http.websocketx.WebSocketFrame;
007    import org.jboss.netty.handler.codec.http.websocketx.WebSocketServerHandshaker;
008    
009    import org.elasticsearch.ElasticsearchException;
010    import org.elasticsearch.common.component.AbstractLifecycleComponent;
011    import org.elasticsearch.common.inject.Inject;
012    import org.elasticsearch.common.settings.Settings;
013    import org.elasticsearch.env.Environment;
014    import org.elasticsearch.node.service.NodeService;
015    import org.elasticsearch.rest.RestController;
016    
017    import org.xbib.elasticsearch.websocket.InteractiveController;
018    import org.xbib.elasticsearch.websocket.Presence;
019    
020    
021    /**
022     * HttpServer with a REST and WebSocket interactive controller.
023     */
024    public class HttpServer extends AbstractLifecycleComponent<HttpServer> {
025    
026        private final Environment environment;
027    
028        private final HttpServerTransport transport;
029    
030        private final RestController restController;
031        
032        private final InteractiveController interActiveController;
033    
034        private final NodeService nodeService;    
035        
036        @Inject
037        public HttpServer(Settings settings, Environment environment, 
038                          HttpServerTransport transport,
039                          RestController restController,
040                          InteractiveController interActiveController,
041                          NodeService nodeService) {
042            super(settings);
043            this.environment = environment;
044            this.transport = transport;
045            this.restController = restController;
046            this.interActiveController = interActiveController;
047            this.nodeService = nodeService;
048    
049            transport.httpServerAdapter(new Dispatcher(this));
050            transport.webSocketServerAdapter(new Interactor(this));
051            
052            // the hook into nodeService can not be used here
053            //nodeService.setHttpServer(this);
054    
055        }
056    
057        static class Dispatcher implements HttpServerAdapter {
058    
059            private final HttpServer server;
060    
061            Dispatcher(HttpServer server) {
062                this.server = server;
063            }
064    
065            @Override
066            public void dispatchRequest(HttpRequest request, HttpChannel channel) {
067                server.internalDispatchRequest(request, channel);
068            }
069        }
070    
071        static class Interactor implements WebSocketServerAdapter {
072    
073            private final HttpServer server;
074    
075            Interactor(HttpServer server) {
076                this.server = server;
077            }
078            
079            @Override
080            public void presence(Presence presence, String topic, Channel channel) {
081                server.internalPresence(presence, topic, channel);
082            }
083    
084            @Override
085            public void frame(WebSocketServerHandshaker handshaker, WebSocketFrame frame, ChannelHandlerContext context) {
086                server.internalFrame(handshaker, frame, context);
087            }
088        }
089    
090        @Override
091        protected void doStart() throws ElasticsearchException {
092            transport.start();
093            logger.info("{}", transport.boundAddress());
094            nodeService.putAttribute("websocket_address", transport.boundAddress().publishAddress().toString());
095        }
096    
097        @Override
098        protected void doStop() throws ElasticsearchException {
099            nodeService.removeAttribute("websocket_address");
100            transport.stop();
101        }
102    
103        @Override
104        protected void doClose() throws ElasticsearchException {
105            transport.close();
106        }
107    
108        public HttpInfo info() {
109            return new HttpInfo(transport.boundAddress());
110        }
111    
112        public HttpStats stats() {
113            return transport.stats();
114        }
115        
116        public Channel channel(Integer id) {
117            return transport.channel(id);
118        }
119    
120        public void internalDispatchRequest(final HttpRequest request, final HttpChannel channel) {
121            restController.dispatchRequest(request, channel);
122        }
123        
124        public void internalPresence(Presence presence, String topic, Channel channel) {
125            interActiveController.presence(presence, topic, channel);
126        }
127        
128        public void internalFrame(WebSocketServerHandshaker handshaker, WebSocketFrame frame, ChannelHandlerContext context) {
129            interActiveController.frame(handshaker, frame, context);   
130        }
131        
132    }
