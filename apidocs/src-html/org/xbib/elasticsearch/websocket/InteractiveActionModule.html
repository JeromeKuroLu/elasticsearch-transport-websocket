002    package org.xbib.elasticsearch.websocket;
003    
004    import java.util.List;
005    
006    import org.elasticsearch.common.collect.Lists;
007    import org.elasticsearch.common.inject.AbstractModule;
008    
009    import org.xbib.elasticsearch.websocket.action.bulk.BulkDeleteAction;
010    import org.xbib.elasticsearch.websocket.action.bulk.BulkFlushAction;
011    import org.xbib.elasticsearch.websocket.action.bulk.BulkIndexAction;
012    import org.xbib.elasticsearch.websocket.action.pubsub.ForwardAction;
013    import org.xbib.elasticsearch.websocket.action.pubsub.PublishAction;
014    import org.xbib.elasticsearch.websocket.action.pubsub.SubscribeAction;
015    import org.xbib.elasticsearch.websocket.action.pubsub.UnsubscribeAction;
016    
017    /**
018     * The InteractiveActionModule binds all WebSocket actions.
019     * 
020     */
021    public class InteractiveActionModule extends AbstractModule {
022    
023        private List<Class<? extends BaseInteractiveHandler>> websocketActions = Lists.newArrayList();
024    
025        public InteractiveActionModule(List<Class<? extends BaseInteractiveHandler>> websocketActions) {
026            this.websocketActions = websocketActions;
027        }
028    
029        @Override
030        protected void configure() {
031            for (Class<? extends BaseInteractiveHandler> websocketAction : websocketActions) {
032                bind(websocketAction).asEagerSingleton();
033            }
034            bind(PublishAction.class).asEagerSingleton();
035            bind(SubscribeAction.class).asEagerSingleton();
036            bind(UnsubscribeAction.class).asEagerSingleton();
037            bind(ForwardAction.class).asEagerSingleton();
038            bind(BulkDeleteAction.class).asEagerSingleton();
039            bind(BulkIndexAction.class).asEagerSingleton();
040            bind(BulkFlushAction.class).asEagerSingleton();
041        }
042    }
