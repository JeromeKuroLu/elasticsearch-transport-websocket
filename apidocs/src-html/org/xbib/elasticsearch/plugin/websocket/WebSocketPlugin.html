001    
002    package org.xbib.elasticsearch.plugin.websocket;
003    
004    import java.util.Collection;
005    
006    import org.elasticsearch.common.component.LifecycleComponent;
007    import org.elasticsearch.common.inject.Module;
008    import org.elasticsearch.common.settings.Settings;
009    import org.elasticsearch.plugins.AbstractPlugin;
010    import org.elasticsearch.rest.RestModule;
011    
012    import static org.elasticsearch.common.collect.Lists.newArrayList;
013    
014    import org.xbib.elasticsearch.websocket.WebSocketModule;
015    import org.xbib.elasticsearch.websocket.action.pubsub.Checkpointer;
016    import org.xbib.elasticsearch.websocket.http.HttpServer;
017    import org.xbib.elasticsearch.websocket.http.HttpServerModule;
018    import org.xbib.elasticsearch.websocket.rest.action.RestPublishAction;
019    import org.xbib.elasticsearch.websocket.rest.action.RestUnsubscribeAction;
020    
021    /**
022     * Websocket plugin
023     */
024    public class WebSocketPlugin extends AbstractPlugin {
025    
026        private final Settings settings;
027    
028        public WebSocketPlugin(Settings settings) {
029            this.settings = settings;
030        }
031    
032        @Override
033        public String name() {
034            return "transport-websocket-"
035                    + Build.getInstance().getVersion() + "-"
036                    + Build.getInstance().getShortHash();
037        }
038    
039        @Override
040        public String description() {
041            return "Websocket transport plugin";
042        }
043    
044        @Override
045        public Collection<Class<? extends Module>> modules() {
046            Collection<Class<? extends Module>> modules = newArrayList();
047            if (settings.getAsBoolean("websocket.enabled", true)) {
048                modules.add(HttpServerModule.class);
049                modules.add(WebSocketModule.class);
050            }
051            return modules;
052        }
053    
054       @Override public Collection<Class<? extends LifecycleComponent>> services() {
055            Collection<Class<? extends LifecycleComponent>> services = newArrayList();
056            if (settings.getAsBoolean("websocket.enabled", true)) {
057                services.add(HttpServer.class);
058                services.add(Checkpointer.class);
059            }
060            return services;
061        }    
062      
063        public void onModule(RestModule module) {
064            module.addRestAction(RestPublishAction.class);
065            module.addRestAction(RestUnsubscribeAction.class);
066        }
067    
068    }
