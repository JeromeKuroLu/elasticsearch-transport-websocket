001package org.xbib.elasticsearch.common.netty;
002
003import org.elasticsearch.common.lease.Releasable;
004import org.jboss.netty.channel.ChannelFuture;
005import org.jboss.netty.channel.ChannelFutureListener;
006
007public class ReleaseChannelFutureListener implements ChannelFutureListener {
008
009    private final Releasable releasable;
010
011    public ReleaseChannelFutureListener(Releasable releasable) {
012        this.releasable = releasable;
013    }
014
015    @Override
016    public void operationComplete(ChannelFuture future) throws Exception {
017        releasable.close();
018    }
019}
