020package org.elasticsearch.websocket.transport.netty;
021
022import java.io.IOException;
023import org.elasticsearch.common.component.Lifecycle;
024import org.elasticsearch.common.io.ThrowableObjectInputStream;
025import org.elasticsearch.common.io.stream.CachedStreamInput;
026import org.elasticsearch.common.io.stream.StreamInput;
027import org.elasticsearch.common.io.stream.Streamable;
028import org.elasticsearch.common.logging.ESLogger;
029import org.elasticsearch.threadpool.ThreadPool;
030import org.elasticsearch.transport.*;
031import org.elasticsearch.transport.support.TransportStreams;
032import org.jboss.netty.buffer.ChannelBuffer;
033import org.jboss.netty.channel.*;
034
035/**
036 * A handler (must be the last one!) that does size based frame decoding and forwards the actual message
037 * to the relevant action.
038 */
039public class MessageChannelHandler extends SimpleChannelUpstreamHandler {
040
041    private final ESLogger logger;
042
043    private final ThreadPool threadPool;
044
045    private final TransportServiceAdapter transportServiceAdapter;
046
047    private final NettyTransport transport;
048
049    public MessageChannelHandler(NettyTransport transport, ESLogger logger) {
050        this.threadPool = transport.threadPool();
051        this.transportServiceAdapter = transport.transportServiceAdapter();
052        this.transport = transport;
053        this.logger = logger;
054    }
055
056    @Override
057    public void writeComplete(ChannelHandlerContext ctx, WriteCompletionEvent e) throws Exception {
058        transportServiceAdapter.sent(e.getWrittenAmount());
059        super.writeComplete(ctx, e);
060    }
061
062    @Override
063    public void messageReceived(ChannelHandlerContext ctx, MessageEvent e) throws Exception {
064        Object m = e.getMessage();
065        if (!(m instanceof ChannelBuffer)) {
066            ctx.sendUpstream(e);
067            return;
068        }
069        ChannelBuffer buffer = (ChannelBuffer) m;
070        int size = buffer.getInt(buffer.readerIndex() - 4);
071
072        transportServiceAdapter.received(size + 4);
073
074        int markedReaderIndex = buffer.readerIndex();
075        int expectedIndexReader = markedReaderIndex + size;
076
077        // netty always copies a buffer, either in NioWorker in its read handler, where it copies to a fresh
078        // buffer, or in the cumlation buffer, which is cleaned each time
079        StreamInput streamIn = ChannelBufferStreamInputFactory.create(buffer, size);
080
081        long requestId = buffer.readLong();
082        byte status = buffer.readByte();
083        boolean isRequest = TransportStreams.statusIsRequest(status);
084
085        // !!! compression handling removed !!!
086        StreamInput wrappedStream = CachedStreamInput.cachedHandles(streamIn);
087
088        if (isRequest) {
089            String action = handleRequest(ctx.getChannel(), wrappedStream, requestId);
090            if (buffer.readerIndex() != expectedIndexReader) {
091                if (buffer.readerIndex() < expectedIndexReader) {
092                    logger.warn("Message not fully read (request) for [{}] and action [{}], resetting", requestId, action);
093                } else {
094                    logger.warn("Message read past expected size (request) for [{}] and action [{}], resetting", requestId, action);
095                }
096                buffer.readerIndex(expectedIndexReader);
097            }
098        } else {
099            TransportResponseHandler handler = transportServiceAdapter.remove(requestId);
100            // ignore if its null, the adapter logs it
101            if (handler != null) {
102                if (TransportStreams.statusIsError(status)) {
103                    handlerResponseError(wrappedStream, handler);
104                } else {
105                    handleResponse(wrappedStream, handler);
106                }
107            } else {
108                // if its null, skip those bytes
109                buffer.readerIndex(markedReaderIndex + size);
110            }
111            if (buffer.readerIndex() != expectedIndexReader) {
112                if (buffer.readerIndex() < expectedIndexReader) {
113                    logger.warn("Message not fully read (response) for [{}] handler {}, error [{}], resetting", requestId, handler, TransportStreams.statusIsError(status));
114                } else {
115                    logger.warn("Message read past expected size (response) for [{}] handler {}, error [{}], resetting", requestId, handler, TransportStreams.statusIsError(status));
116                }
117                buffer.readerIndex(expectedIndexReader);
118            }
119        }
120        wrappedStream.close();
121    }
122
123    private void handleResponse(StreamInput buffer, final TransportResponseHandler handler) {
124        final Streamable streamable = handler.newInstance();
125        try {
126            streamable.readFrom(buffer);
127        } catch (Exception e) {
128            handleException(handler, new TransportSerializationException("Failed to deserialize response of type [" + streamable.getClass().getName() + "]", e));
129            return;
130        }
131        try {
132            if (handler.executor() == ThreadPool.Names.SAME) {
133                //noinspection unchecked
134                handler.handleResponse(streamable);
135            } else {
136                threadPool.executor(handler.executor()).execute(new ResponseHandler(handler, streamable));
137            }
138        } catch (Exception e) {
139            handleException(handler, new ResponseHandlerFailureTransportException(e));
140        }
141    }
142
143    private void handlerResponseError(StreamInput buffer, final TransportResponseHandler handler) {
144        Throwable error;
145        try {
146            ThrowableObjectInputStream ois = new ThrowableObjectInputStream(buffer, transport.settings().getClassLoader());
147            error = (Throwable) ois.readObject();
148        } catch (Exception e) {
149            error = new TransportSerializationException("Failed to deserialize exception response from stream", e);
150        }
151        handleException(handler, error);
152    }
153
154    private void handleException(final TransportResponseHandler handler, Throwable error) {
155        if (!(error instanceof RemoteTransportException)) {
156            error = new RemoteTransportException(error.getMessage(), error);
157        }
158        final RemoteTransportException rtx = (RemoteTransportException) error;
159        if (handler.executor() == ThreadPool.Names.SAME) {
160            handler.handleException(rtx);
161        } else {
162            threadPool.executor(handler.executor()).execute(new Runnable() {
163                @Override
164                public void run() {
165                    try {
166                        handler.handleException(rtx);
167                    } catch (Exception e) {
168                        logger.error("Failed to handle exception response", e);
169                    }
170                }
171            });
172        }
173    }
174
175    private String handleRequest(Channel channel, StreamInput buffer, long requestId) throws IOException {
176        final String action = buffer.readUTF();
177
178        final NettyTransportChannel transportChannel = new NettyTransportChannel(transport, action, channel, requestId);
179        try {
180            final TransportRequestHandler handler = transportServiceAdapter.handler(action);
181            if (handler == null) {
182                throw new ActionNotFoundTransportException(action);
183            }
184            final Streamable streamable = handler.newInstance();
185            streamable.readFrom(buffer);
186            if (handler.executor() == ThreadPool.Names.SAME) {
187                //noinspection unchecked
188                handler.messageReceived(streamable, transportChannel);
189            } else {
190                threadPool.executor(handler.executor()).execute(new RequestHandler(handler, streamable, transportChannel, action));
191            }
192        } catch (Exception e) {
193            try {
194                transportChannel.sendResponse(e);
195            } catch (IOException e1) {
196                logger.warn("Failed to send error message back to client for action [" + action + "]", e);
197                logger.warn("Actual Exception", e1);
198            }
199        }
200        return action;
201    }
202
203    @Override
204    public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e) throws Exception {
205        transport.exceptionCaught(ctx, e);
206    }
207
208    class ResponseHandler implements Runnable {
209
210        private final TransportResponseHandler handler;
211        private final Streamable streamable;
212
213        public ResponseHandler(TransportResponseHandler handler, Streamable streamable) {
214            this.handler = handler;
215            this.streamable = streamable;
216        }
217
218        @SuppressWarnings({"unchecked"})
219        @Override
220        public void run() {
221            try {
222                handler.handleResponse(streamable);
223            } catch (Exception e) {
224                handleException(handler, new ResponseHandlerFailureTransportException(e));
225            }
226        }
227    }
228
229    class RequestHandler implements Runnable {
230        private final TransportRequestHandler handler;
231        private final Streamable streamable;
232        private final NettyTransportChannel transportChannel;
233        private final String action;
234
235        public RequestHandler(TransportRequestHandler handler, Streamable streamable, NettyTransportChannel transportChannel, String action) {
236            this.handler = handler;
237            this.streamable = streamable;
238            this.transportChannel = transportChannel;
239            this.action = action;
240        }
241
242        @SuppressWarnings({"unchecked"})
243        @Override
244        public void run() {
245            try {
246                handler.messageReceived(streamable, transportChannel);
247            } catch (Throwable e) {
248                if (transport.lifecycleState() == Lifecycle.State.STARTED) {
249                    // we can only send a response transport is started....
250                    try {
251                        transportChannel.sendResponse(e);
252                    } catch (IOException e1) {
253                        logger.warn("Failed to send error message back to client for action [" + action + "]", e1);
254                        logger.warn("Actual Exception", e);
255                    }
256                }
257            }
258        }
259    }
