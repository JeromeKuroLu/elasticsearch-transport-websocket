020package org.elasticsearch.websocket.http.netty;
021
022import org.jboss.netty.channel.ChannelHandlerContext;
023import org.jboss.netty.channel.ExceptionEvent;
024import org.jboss.netty.channel.MessageEvent;
025import org.jboss.netty.handler.codec.http.HttpRequest;
026import org.jboss.netty.handler.codec.http.websocketx.WebSocketFrame;
027
028/**
029 * Netty implementation for a WebSocket request handler.
030 * It is based on the HTTP request handler.
031 * 
032 * @author JÃ¶rg Prante <joergprante@gmail.com>
033 */
034public class NettyWebSocketRequestHandler extends NettyHttpRequestHandler {
035    
036   public NettyWebSocketRequestHandler(NettyWebSocketServerTransport serverTransport) {
037        super(serverTransport);
038    }
039    
040    @Override
041    public void messageReceived(ChannelHandlerContext ctx, MessageEvent e) throws Exception {
042        Object msg = e.getMessage();
043        if (msg instanceof HttpRequest) {
044            super.messageReceived(ctx, e);            
045        } else if (msg instanceof WebSocketFrame) {
046            serverTransport.frame(handshaker, (WebSocketFrame) msg, ctx);
047        }
048    }
049
050    @Override
051    public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e) throws Exception {
052        serverTransport.exceptionCaught(ctx, e);
053        e.getChannel().close();
054    }
055
056}
