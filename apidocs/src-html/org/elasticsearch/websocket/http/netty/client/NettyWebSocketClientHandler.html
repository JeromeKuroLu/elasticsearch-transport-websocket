019package org.elasticsearch.websocket.http.netty.client;
020
021import java.net.InetSocketAddress;
022import java.net.URI;
023import org.elasticsearch.websocket.client.WebSocketActionListener;
024import org.elasticsearch.websocket.client.WebSocketClient;
025import org.jboss.netty.bootstrap.ClientBootstrap;
026import org.jboss.netty.channel.Channel;
027import org.jboss.netty.channel.ChannelFuture;
028import org.jboss.netty.channel.ChannelHandlerContext;
029import org.jboss.netty.channel.ChannelStateEvent;
030import org.jboss.netty.channel.ExceptionEvent;
031import org.jboss.netty.channel.MessageEvent;
032import org.jboss.netty.channel.SimpleChannelUpstreamHandler;
033import org.jboss.netty.handler.codec.http.DefaultHttpRequest;
034import org.jboss.netty.handler.codec.http.HttpHeaders.Names;
035import org.jboss.netty.handler.codec.http.HttpHeaders.Values;
036import org.jboss.netty.handler.codec.http.HttpMethod;
037import org.jboss.netty.handler.codec.http.HttpRequest;
038import org.jboss.netty.handler.codec.http.HttpResponse;
039import org.jboss.netty.handler.codec.http.HttpResponseStatus;
040import org.jboss.netty.handler.codec.http.HttpVersion;
041import org.jboss.netty.handler.codec.http.websocketx.WebSocket00FrameDecoder;
042import org.jboss.netty.handler.codec.http.websocketx.WebSocket00FrameEncoder;
043import org.jboss.netty.handler.codec.http.websocketx.WebSocketFrame;
044import org.jboss.netty.util.CharsetUtil;
045
046/**
047 * Handles socket communication for a connected WebSocket client. Not intended
048 * for end-users. Please use {@link NettyWebSocketClient} or
049 * {@link NettyWebSocketActionListener} for controlling your client.
050 *
051 * @author <a href="http://www.pedantique.org/">Carl Bystr&ouml;m</a>
052 */
053public class NettyWebSocketClientHandler extends SimpleChannelUpstreamHandler
054        implements WebSocketClient {
055
056    private final ClientBootstrap bootstrap;
057    private final URI url;
058    private final WebSocketActionListener listener;
059    private boolean handshakeCompleted = false;
060    private Channel channel;
061
062    public NettyWebSocketClientHandler(ClientBootstrap bootstrap, URI url, WebSocketActionListener listener) {
063        this.bootstrap = bootstrap;
064        this.url = url;
065        this.listener = listener;
066    }
067    
068    @Override
069    public Channel channel() {
070        return channel;
071    }
072
073    @Override
074    public void channelConnected(ChannelHandlerContext ctx, ChannelStateEvent event) throws Exception {
075        String path = url.getPath();
076        if (url.getQuery() != null && url.getQuery().length() > 0) {
077            path = url.getPath() + "?" + url.getQuery();
078        }
079        HttpRequest request = new DefaultHttpRequest(HttpVersion.HTTP_1_1, HttpMethod.GET, path);
080        request.addHeader(Names.UPGRADE, Values.WEBSOCKET);
081        request.addHeader(Names.CONNECTION, Values.UPGRADE);
082        request.addHeader(Names.HOST, url.getHost());
083        request.addHeader(Names.ORIGIN, "http://" + url.getHost());
084        event.getChannel().write(request);
085        ctx.getPipeline().replace("encoder", "ws-encoder", new WebSocket00FrameEncoder());
086        this.channel = event.getChannel();
087    }
088
089    @Override
090    public void channelClosed(ChannelHandlerContext ctx, ChannelStateEvent event) throws Exception {
091        listener.onDisconnect(this);
092        handshakeCompleted = false;
093        channel = null;
094    }
095
096    @Override
097    public void messageReceived(ChannelHandlerContext ctx, MessageEvent event) throws Exception {
098        if (!handshakeCompleted) {
099            HttpResponse response = (HttpResponse) event.getMessage();
100            final HttpResponseStatus status = new HttpResponseStatus(101, "Web Socket Protocol Handshake");
101            final boolean validStatus = response.getStatus().equals(status);
102            final boolean validUpgrade = response.getHeader(Names.UPGRADE).equals(Values.WEBSOCKET);
103            final boolean validConnection = response.getHeader(Names.CONNECTION).equals(Values.UPGRADE);
104            if (!validStatus || !validUpgrade || !validConnection) {
105                throw new NettyWebSocketException("Invalid handshake response");
106            }
107            handshakeCompleted = true;
108            ctx.getPipeline().replace("decoder", "ws-decoder", new WebSocket00FrameDecoder());
109            listener.onConnect(this);
110            return;
111        }
112        if (event.getMessage() instanceof HttpResponse) {
113            HttpResponse response = (HttpResponse) event.getMessage();
114            throw new NettyWebSocketException("Unexpected HttpResponse (status=" + response.getStatus() + ", content=" + response.getContent().toString(CharsetUtil.UTF_8) + ")");
115        }
116        WebSocketFrame frame = (WebSocketFrame) event.getMessage();
117        listener.onMessage(this, frame);
118    }
119
120    @Override
121    public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e) throws Exception {
122        final Throwable t = e.getCause();
123        listener.onError(t);
124        e.getChannel().close();
125    }
126
127    @Override
128    public ChannelFuture connect() {
129        return bootstrap.connect(new InetSocketAddress(url.getHost(), url.getPort()));
130    }
131
132    @Override
133    public ChannelFuture disconnect() {
134        if (channel == null) {
135            return null;
136        }
137        return channel.close();
138    }
139
140    @Override
141    public ChannelFuture send(WebSocketFrame frame) {
142        if (channel == null) {
143            return null;
144        }
145        return channel.write(frame);
146    }
147
148}
