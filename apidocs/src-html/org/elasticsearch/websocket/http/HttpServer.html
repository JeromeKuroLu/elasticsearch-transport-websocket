020package org.elasticsearch.websocket.http;
021
022import org.elasticsearch.ElasticSearchException;
023import org.elasticsearch.common.component.AbstractLifecycleComponent;
024import org.elasticsearch.common.inject.Inject;
025import org.elasticsearch.common.settings.Settings;
026import org.elasticsearch.env.Environment;
027import org.elasticsearch.node.service.NodeService;
028import org.elasticsearch.rest.RestController;
029import org.elasticsearch.websocket.InteractiveController;
030import org.elasticsearch.websocket.Presence;
031import org.jboss.netty.channel.Channel;
032import org.jboss.netty.channel.ChannelHandlerContext;
033import org.jboss.netty.handler.codec.http.websocketx.WebSocketFrame;
034import org.jboss.netty.handler.codec.http.websocketx.WebSocketServerHandshaker;
035
036/**
037 * HttpServer with a REST and WebSocket interactive controller.
038 */
039public class HttpServer extends AbstractLifecycleComponent<HttpServer> {
040
041    private final Environment environment;
042
043    private final HttpServerTransport transport;
044
045    private final RestController restController;
046    
047    private final InteractiveController interActiveController;
048
049    private final NodeService nodeService;    
050    
051    @Inject
052    public HttpServer(Settings settings, Environment environment, 
053                      HttpServerTransport transport,
054                      RestController restController,
055                      InteractiveController interActiveController,
056                      NodeService nodeService) {
057        super(settings);
058        this.environment = environment;
059        this.transport = transport;
060        this.restController = restController;
061        this.interActiveController = interActiveController;
062        this.nodeService = nodeService;
063
064        transport.httpServerAdapter(new Dispatcher(this));
065        transport.webSocketServerAdapter(new Interactor(this));
066        
067        // the hook into nodeService is used for optional node info 
068        // (not usable with relocated HttpServer class)
069        
070        //nodeService.setHttpServer(this);
071
072    }
073
074    static class Dispatcher implements HttpServerAdapter {
075
076        private final HttpServer server;
077
078        Dispatcher(HttpServer server) {
079            this.server = server;
080        }
081
082        @Override
083        public void dispatchRequest(HttpRequest request, HttpChannel channel) {
084            server.internalDispatchRequest(request, channel);
085        }
086    }
087
088    static class Interactor implements WebSocketServerAdapter {
089
090        private final HttpServer server;
091
092        Interactor(HttpServer server) {
093            this.server = server;
094        }
095        
096        @Override
097        public void presence(Presence presence, String topic, Channel channel) {
098            server.internalPresence(presence, topic, channel);
099        }
100
101        @Override
102        public void frame(WebSocketServerHandshaker handshaker, WebSocketFrame frame, ChannelHandlerContext context) {
103            server.internalFrame(handshaker, frame, context);
104        }
105    }
106
107    @Override
108    protected void doStart() throws ElasticSearchException {
109        transport.start();
110        logger.info("{}", transport.boundAddress());
111        nodeService.putAttribute("websocket_address", transport.boundAddress().publishAddress().toString());
112    }
113
114    @Override
115    protected void doStop() throws ElasticSearchException {
116        nodeService.removeAttribute("websocket_address");
117        transport.stop();
118    }
119
120    @Override
121    protected void doClose() throws ElasticSearchException {
122        transport.close();
123    }
124
125    public HttpInfo info() {
126        return new HttpInfo(transport.boundAddress());
127    }
128
129    public HttpStats stats() {
130        return transport.stats();
131    }
132    
133    public Channel channel(Integer id) {
134        return transport.channel(id);
135    }
136
137    public void internalDispatchRequest(final HttpRequest request, final HttpChannel channel) {
138        restController.dispatchRequest(request, channel);
139    }
140    
141    public void internalPresence(Presence presence, String topic, Channel channel) {
142        interActiveController.presence(presence, topic, channel);
143    }
144    
145    public void internalFrame(WebSocketServerHandshaker handshaker, WebSocketFrame frame, ChannelHandlerContext context) {
146        interActiveController.frame(handshaker, frame, context);   
147    }
148    
149}
