020package org.elasticsearch.websocket;
021
022import java.util.HashMap;
023import java.util.Map;
024import org.elasticsearch.ElasticSearchException;
025import org.elasticsearch.common.component.AbstractLifecycleComponent;
026import org.elasticsearch.common.inject.Inject;
027import org.elasticsearch.common.settings.Settings;
028import org.elasticsearch.common.xcontent.XContentFactory;
029import org.elasticsearch.common.xcontent.XContentParser;
030import org.elasticsearch.websocket.http.netty.NettyInteractiveChannel;
031import org.elasticsearch.websocket.http.netty.NettyInteractiveRequest;
032import org.jboss.netty.channel.Channel;
033import org.jboss.netty.channel.ChannelHandlerContext;
034import org.jboss.netty.handler.codec.http.websocketx.CloseWebSocketFrame;
035import org.jboss.netty.handler.codec.http.websocketx.PingWebSocketFrame;
036import org.jboss.netty.handler.codec.http.websocketx.PongWebSocketFrame;
037import org.jboss.netty.handler.codec.http.websocketx.TextWebSocketFrame;
038import org.jboss.netty.handler.codec.http.websocketx.WebSocketFrame;
039import org.jboss.netty.handler.codec.http.websocketx.WebSocketServerHandshaker;
040
041/**
042 * The InteractiveController controls the presence of websocket connections
043 * and the flow of websocket frames for interactive use.
044 * 
045 * @author JÃ¶rg Prante <joergprante@gmail.com>
046 */
047public class InteractiveController extends AbstractLifecycleComponent<InteractiveController> {
048
049    private final HashMap<String, InteractiveHandler> handlers = new HashMap();
050
051    @Inject
052    public InteractiveController(Settings settings) {
053        super(settings);
054    }
055
056    @Override
057    protected void doStart() throws ElasticSearchException {
058    }
059
060    @Override
061    protected void doStop() throws ElasticSearchException {
062    }
063
064    @Override
065    protected void doClose() throws ElasticSearchException {
066    }
067
068    public void registerHandler(String type, InteractiveHandler handler) {
069        handlers.put(type, handler);
070    }
071
072    public void presence(Presence presence, String topic, Channel channel) {
073        if (logger.isDebugEnabled()) {
074            logger.debug("presence: " + presence.name()
075                    + " topic =" + topic
076                    + " channel =" + channel);
077        }
078    }
079
080    public void frame(WebSocketServerHandshaker handshaker, WebSocketFrame frame, ChannelHandlerContext context) {
081        if (frame instanceof TextWebSocketFrame) {
082            text((TextWebSocketFrame) frame, context.getChannel());
083        } else if (handshaker != null && frame instanceof CloseWebSocketFrame) {
084            handshaker.close(context.getChannel(), (CloseWebSocketFrame) frame);
085            presence(Presence.DISCONNECTED, null, context.getChannel());
086        } else if (frame instanceof PingWebSocketFrame) {
087            context.getChannel().write(new PongWebSocketFrame(frame.getBinaryData()));
088        }
089    }
090
091    private void text(TextWebSocketFrame frame, Channel channel) {
092        Map<String, Object> map = parse(frame.getText());
093        if (map == null) {
094            error("invalid request", channel);
095            return;
096        }
097        String type = (String) map.get("type");
098        if (type == null) {
099            error("no type found", channel);
100            return;
101        }
102        if (!handlers.containsKey(type)) {
103            error("missing handler for type " + type, channel);
104            return;
105        }
106        Map<String, Object> data = (Map<String, Object>) map.get("data");
107        handlers.get(type).handleRequest(new NettyInteractiveRequest(data), new NettyInteractiveChannel(channel));
108    }
109
110    private Map<String, Object> parse(String source) {
111        Map<String, Object> map = null;
112        try (XContentParser parser = XContentFactory.xContent(source).createParser(source)) {
113            map = parser.map();
114        } catch (Exception e) {
115            logger.warn("unable to parse {}", source);
116        }
117        return map;
118    }
119
120    private void error(String message, Channel channel) {
121        TextWebSocketFrame frame = new TextWebSocketFrame();
122        String text = "{\"ok\":false,\"error\":\"" + message + "\"}";
123        frame.setText(text);
124        channel.write(frame);
125    }
126}
